```python
# === CREATE FILE: backend/services/rbac.py ===

from fastapi import HTTPException, Request
from . import core  # Import shared core module

class RBACService:
    """Centralized permission evaluation with team/data/action scoping"""
    
    @staticmethod
    async def check_permission(
        user: dict,
        action: core.ActionType,
        data_type: core.DataType,
        resource_context: dict,  # {station_id, team_id, owner_id, etc.}
        request: Request
    ) -> bool:
        """
        Evaluate if user can perform action on data_type given resource context.
        Returns True if authorized, raises 403 otherwise.
        """
        # Founder bypass (global admin)
        if user.get("email") == core.FOUNDER_EMAIL:
            return True
            
        # Guest users: read-only on public data
        if user.get("is_guest"):
            if action == core.ActionType.READ and data_type in {
                core.DataType.STATION, core.DataType.INVENTORY
            }:
                return resource_context.get("is_public", False)
            return False
        
        # Load permissions if not cached
        permissions = user.get("permissions", [])
        if not permissions and user.get("id"):
            permissions = await RBACService._load_user_permissions(user["id"])
            user["permissions"] = permissions
        
        # Evaluate each permission rule
        for perm in permissions:
            if RBACService._matches_rule(perm, action, data_type, resource_context, user):
                return True
                
        raise HTTPException(
            status_code=403,
            detail=f"Permission denied: {action.value} on {data_type.value}",
            headers={"X-Permission-Debug": f"User:{user.get('id')}|Scope:{user.get('team_scope')}"}
        )
    
    @staticmethod
    def _matches_rule(
        perm: core.Permission,
        action: core.ActionType,
        data_type: core.DataType,
        context: dict,
        user: dict
    ) -> bool:
        """Check if a single permission rule matches the request"""
        # Action and data type must match exactly
        if perm.action != action or perm.data_type != data_type:
            return False
            
        # Team scope evaluation
        if perm.team_scope == core.TeamScope.GLOBAL:
            return user.get("tier") in {"pro", "enterprise"}
            
        elif perm.team_scope == core.TeamScope.STATION:
            # Must have station_id in context AND in user's assigned stations
            ctx_station = context.get("station_id") or context.get("resource_station_id")
            return (
                ctx_station 
                and perm.station_id == ctx_station
                and ctx_station in user.get("assigned_stations", [])
            )
            
        elif perm.team_scope == core.TeamScope.TEAM:
            # Check team membership
            return (
                user.get("team_id") 
                and user["team_id"] == context.get("team_id")
            )
            
        elif perm.team_scope == core.TeamScope.PERSONAL:
            # Owner check
            return context.get("owner_id") == user.get("id")
            
        return False
    
    @staticmethod
    async def _load_user_permissions(user_id: str) -> list[core.Permission]:
        """Load permissions from DB based on user role/tier/station assignments"""
        from core import db
        
        user_doc = await db.users.find_one({"id": user_id}, {"_id": 0})
        if not user_doc:
            return []
            
        permissions = []
        role = user_doc.get("role", "staff")
        tier = user_doc.get("tier", "free")
        stations = user_doc.get("assigned_stations", [])
        
        # Role-based permission templates
        if role == "owner":
            permissions.extend([
                core.Permission(action=a, data_type=d, team_scope=core.TeamScope.STATION, station_id=s)
                for a in core.ActionType for d in core.DataType
                for s in stations or [None]  # All assigned stations
            ])
        elif role == "manager":
            permissions.extend([
                core.Permission(action=a, data_type=d, team_scope=core.TeamScope.STATION, station_id=s)
                for a in [core.ActionType.READ, core.ActionType.CREATE, core.ActionType.UPDATE]
                for d in [core.DataType.SALE, core.DataType.INVENTORY, core.DataType.EMPLOYEE]
                for s in stations
            ])
        elif role == "staff":
            permissions.extend([
                core.Permission(action=core.ActionType.READ, data_type=d, team_scope=core.TeamScope.STATION, station_id=s)
                for d in [core.DataType.SALE, core.DataType.INVENTORY]
                for s in stations
            ])
            permissions.append(core.Permission(
                action=core.ActionType.CREATE, 
                data_type=core.DataType.SALE, 
                team_scope=core.TeamScope.PERSONAL
            ))
        elif role == "auditor":
            permissions.append(core.Permission(
                action=core.ActionType.READ, 
                data_type=core.DataType.AUDIT_LOG, 
                team_scope=core.TeamScope.GLOBAL
            ))
            permissions.extend([
                core.Permission(action=core.ActionType.READ, data_type=d, team_scope=core.TeamScope.STATION, station_id=s)
                for d in [core.DataType.SALE, core.DataType.INVENTORY, core.DataType.EXPENSE]
                for s in stations
            ])
            
        return permissions
```

```python
async def require_permission(
    action: core.ActionType,

    current_user: dict = Depends(core.get_current_user)
):
    """
    FastAPI dependency to enforce RBAC at route level.

    }
    
    await rbac.RBACService.check_permission(
        user=current_user,
        action=action,
        data_type=data_type,
        resource_context=resource_context,
        request=request
    )
    return current_user  # Return user for route handler use
```

```python
# === CREATE FILE: backend/services/audit.py ===

import hashlib
import hmac
import json
from .. import core

class AuditService:
    """SOC-2 compliant audit logging with immutability guarantees"""
    
    # Server-side secret for log signing (rotate quarterly)
    _AUDIT_HMAC_KEY = core.os.environ.get("AUDIT_HMAC_KEY", core.JWT_SECRET)
    
    @classmethod
    def _compute_log_hash(cls, entry: dict) -> str:
        """Compute SHA-256 hash of critical fields for tamper detection"""
        critical = {
            "user_id": entry["user_id"],
            "action": entry["action"],
            "resource_type": entry["resource_type"],
            "resource_id": entry.get("resource_id"),
            "timestamp": entry["timestamp"],
            "ip_address": entry["ip_address"]
        }
        return hashlib.sha256(
            json.dumps(critical, sort_keys=True).encode()
        ).hexdigest()
    
    @classmethod
    def _compute_signature(cls, entry: dict) -> str:
        """HMAC signature for integrity verification"""
        message = json.dumps({
            "user_id": entry["user_id"],
            "action": entry["action"],
            "resource_type": entry["resource_type"],
            "timestamp": entry["timestamp"],
            "log_hash": cls._compute_log_hash(entry)
        }, sort_keys=True)
        return hmac.new(
            cls._AUDIT_HMAC_KEY.encode(),
            message.encode(),
            hashlib.sha256
        ).hexdigest()
    
    @classmethod
    async def log_access(
        cls,
        user_id: str,
        action: str,
        resource_type: str,
        resource_id: Optional[str] = None,
        station_id: Optional[str] = None,
        ip_address: str = "unknown",
        user_agent: str = "unknown",
        user_email: str = "unknown",
        user_role: str = "unknown",
        request_payload: Optional[dict] = None,
        before_state: Optional[dict] = None,
        after_state: Optional[dict] = None,
        reason: Optional[str] = None
    ):
        """Write an immutable audit log entry"""
        
        # Build base entry
        entry = core.AuditLogEntry(
            user_id=user_id,
            user_email=user_email,
            user_role=user_role,
            action=action,
            resource_type=resource_type,
            resource_id=resource_id,
            resource_snapshot_before=before_state,
            resource_snapshot_after=after_state,
            ip_address=ip_address,
            user_agent=user_agent,
            station_id=station_id,
        )
        
        # Hash request payload if provided (never store raw sensitive data)
        if request_payload:
            # Remove sensitive fields before hashing
            sanitized = {k: v for k, v in request_payload.items() 
                        if k not in {"password", "token", "secret", "card_number"}}
            entry.request_payload_hash = hashlib.sha256(
                json.dumps(sanitized, sort_keys=True, default=str).encode()
            ).hexdigest()
        
        # Chain to previous log for tamper-evident sequencing
        last_log = await core.db[core.AUDIT_COLLECTION].find_one(
            {}, sort=[("timestamp", -1)], projection={"log_hash": 1}
        )
        if last_log:
            entry.previous_log_hash = last_log.get("log_hash")
        
        # Compute integrity fields
        entry.log_signature = cls._compute_signature(entry.model_dump())
        
        # Write to SEPARATE audit collection (never mixed with app data)
        audit_doc = entry.model_dump(mode="json")
        audit_doc["log_hash"] = cls._compute_log_hash(audit_doc)  # For indexing
        
        # Insert with write concern for durability
        await core.db[core.AUDIT_COLLECTION].insert_one(
            audit_doc,
            # MongoDB options for SOC-2:
            # - journal: true (write-ahead logging)
            # - w: "majority" (replica set acknowledgment)
        )
        
        return entry.id
    
    @classmethod
    async def query_logs(
        cls,
        *,
        user_id: Optional[str] = None,
        action: Optional[str] = None,
        resource_type: Optional[str] = None,
        station_id: Optional[str] = None,
        start_time: Optional[str] = None,
        end_time: Optional[str] = None,
        limit: int = 100,
        requester_user: dict  # For access control on audit queries
    ) -> list[dict]:
        """
        Query audit logs with RBAC enforcement.
        Only auditors/founders can access full logs; managers see station-scoped logs.
        """
        from .rbac import RBACService
        
        # Enforce: only auditors/founders can query audit logs
        await RBACService.check_permission(
            user=requester_user,
            action=core.ActionType.READ,
            data_type=core.DataType.AUDIT_LOG,
            resource_context={},
            request=None  # Not needed for service-layer calls
        )
        
        # Build query with scoping
        query = {}
        if station_id and requester_user.get("role") != "founder":
            # Non-founders can only see logs for their stations
            query["station_id"] = {"$in": requester_user.get("assigned_stations", [])}
        if user_id:
            query["user_id"] = user_id
        if action:
            query["action"] = action
        if resource_type:
            query["resource_type"] = resource_type
        if start_time or end_time:
            query["timestamp"] = {}
            if start_time: query["timestamp"]["$gte"] = start_time
            if end_time: query["timestamp"]["$lte"] = end_time
        
        # Execute query with projection (exclude sensitive snapshots by default)
        projection = {
            "resource_snapshot_before": 0,  # Exclude by default
            "resource_snapshot_after": 0,
            "request_payload_hash": 1,  # Include hash for verification
            "log_signature": 1,
            "log_hash": 1
        }
        
        cursor = core.db[core.AUDIT_COLLECTION].find(
            query,
            projection=projection,
            sort=[("timestamp", -1)],
            limit=limit
        )
        
        return [core._strip_oid(doc) async for doc in cursor]
    
    @classmethod
    async def verify_log_integrity(cls, log_id: str) -> dict:
        """Verify a log entry hasn't been tampered with"""
        log = await core.db[core.AUDIT_COLLECTION].find_one({"id": log_id})
        if not log:
            return {"valid": False, "error": "Log not found"}
        
        # Recompute hash and signature
        expected_hash = cls._compute_log_hash(log)
        expected_sig = cls._compute_signature(log)
        
        # Verify chain if previous hash exists
        chain_valid = True
        if log.get("previous_log_hash"):
            prev_log = await core.db[core.AUDIT_COLLECTION].find_one(
                {"log_hash": log["previous_log_hash"]}
            )
            chain_valid = prev_log is not None
        
        return {
            "valid": (
                log.get("log_hash") == expected_hash and
                log.get("log_signature") == expected_sig and
                chain_valid
            ),
            "hash_match": log.get("log_hash") == expected_hash,
            "signature_match": log.get("log_signature") == expected_sig,
            "chain_valid": chain_valid
        }
```

```python
from fastapi import Request, Response
from starlette.middleware.base import BaseHTTPMiddleware
from ..services import audit, core

class AuditLoggingMiddleware(BaseHTTPMiddleware):
    """Automatically log all API requests for SOC-2 compliance"""
    
    async def dispatch(self, request: Request, call_next):
        # Capture request start
        start_time = core.now_iso()
        request_id = core.new_id()
        request.state.audit_request_id = request_id
        
        # Get user context (may be None for public endpoints)
        user = await core.get_current_user_optional(
            creds=await core.bearer(request)
        )
        
        # Execute request
        response: Response = await call_next(request)
        
        # Skip logging for health checks and static assets
        if request.url.path in {"/health", "/metrics", "/docs", "/redoc"}:
            return response
            
        # Determine action from HTTP method + path
        action_map = {
            "GET": "read",
            "POST": "create", 
            "PUT": "update",
            "PATCH": "update",
            "DELETE": "delete"
        }
        action = action_map.get(request.method, "unknown")
        
        # Extract resource context from path
        path_parts = request.url.path.strip("/").split("/")
        resource_type = path_parts[-2] if len(path_parts) >= 2 else "unknown"
        resource_id = path_parts[-1] if len(path_parts) >= 3 and path_parts[-2] != "api" else None
        
        # Log asynchronously (non-blocking)
        core.asyncio.create_task(
            audit.AuditService.log_access(
                user_id=user["id"] if user else "anonymous",
                user_email=user.get("email", "anonymous") if user else "anonymous",
                user_role=user.get("role", "guest") if user else "guest",
                action=action,
                resource_type=resource_type,
                resource_id=resource_id,
                station_id=request.headers.get("x-station-id"),
                ip_address=request.client.host if request.client else "unknown",
                user_agent=request.headers.get("user-agent", "unknown"),
                request_payload=await self._safe_read_body(request),
            )
        )
        
        return response
    
    @staticmethod
    async def _safe_read_body(request: Request) -> Optional[dict]:
        """Read request body without consuming stream (for logging)"""
        try:
            # Note: This requires body cache middleware in production
            # For now, log only for mutation endpoints with known schemas
            if request.method in {"POST", "PUT", "PATCH"} and "application/json" in request.headers.get("content-type", ""):
                body = await request.json()
                # Sanitize before logging
                return {k: v for k, v in body.items() if k not in {"password", "token"}}
        except:
            pass
        return None
```

```python
# === CREATE FILE: backend/services/isolation.py ===

from fastapi import HTTPException
from .. import core

class IsolationService:
    """Enforce data isolation at query layer"""
    
    @classmethod
    def apply_isolation_filter(
        cls,
        base_query: dict,
        user: dict,
        resource_type: str,
        requested_station_id: Optional[str] = None
    ) -> dict:
        """
        Inject mandatory isolation filters into MongoDB queries.
        Prevents cross-tenant data access at the database level.
        """
        # Founder/enterprise global access bypass (auditable)
        if user.get("email") == core.FOUNDER_EMAIL:
            return base_query
            
        # Guest users: only public data
        if user.get("is_guest"):
            return {**base_query, "is_public": True}
        
        # Logical isolation: enforce station_id scoping
        assigned_stations = user.get("assigned_stations", [])
        if not assigned_stations and user.get("role") != "founder":
            # Users must have at least one station assignment (except founder)
            raise HTTPException(
                403, 
                "Access denied: user has no assigned stations",
                headers={"X-Isolation-Error": "no_station_assignment"}
            )
        
        # If specific station requested, verify it's in user's assignments
        if requested_station_id:
            if requested_station_id not in assigned_stations:
                raise HTTPException(
                    403,
                    f"Access denied: station {requested_station_id} not assigned to user",
                    headers={"X-Isolation-Error": "station_scope_violation"}
                )
            # Filter to exact station
            return {**base_query, "station_id": requested_station_id}
        
        # Default: allow access to all assigned stations
        return {
            **base_query,
            "station_id": {"$in": assigned_stations}
        }
    
    @classmethod
    async def get_isolated_collection(
        cls,
        collection_name: str,
        user: dict,
        station_id: Optional[str] = None
    ):
        """
        Return MongoDB collection with isolation applied.
        For enterprise tenants in physical mode, route to dedicated DB.
        """
        # Physical isolation for enterprise tenants
        if (
            ISOLATION_CONFIG.mode in {"physical", "hybrid"}
            and user.get("tier") == "enterprise"
            and user.get("isolation_db_name")
        ):
            # Route to dedicated database
            client = core.AsyncIOMotorClient(core.MONGO_URL)
            return client[user["isolation_db_name"]][collection_name]
        
        # Default: logical isolation on shared DB
        return core.db[collection_name]
    
    @classmethod
    def validate_resource_ownership(
        cls,
        resource: dict,
        user: dict,
        resource_type: str
    ) -> bool:
        """Verify resource belongs to user's scope before returning"""
        if user.get("email") == core.FOUNDER_EMAIL:
            return True
            
        resource_station = resource.get("station_id")
        if not resource_station:
            # Resources without station_id are considered public/global
            return resource.get("is_public", False)
            
        return resource_station in user.get("assigned_stations", [])
```

```python
from pydantic import BaseModel
from ... import core
from .. import isolation

T = TypeVar("T", bound=BaseModel)

class IsolatedRepository(Generic[T]):
    """Base repository enforcing data isolation on all operations"""
    
    collection_name: str
    model_class: type[T]
    
    def __init__(self, user: dict):
        self.user = user
    
    async def _get_collection(self, station_id: Optional[str] = None):
        """Get collection with isolation applied"""
        return await isolation.IsolationService.get_isolated_collection(
            self.collection_name,
            self.user,
            station_id
        )
    
    async def find_one(
        self, 
        query: dict, 
        station_id: Optional[str] = None,
        projection: Optional[dict] = None
    ) -> Optional[T]:
        """Find with automatic isolation filter"""
        isolated_query = isolation.IsolationService.apply_isolation_filter(
            query, self.user, self.collection_name, station_id
        )
        
        coll = await self._get_collection(station_id)
        doc = await coll.find_one(isolated_query, projection=projection)
        
        if not doc:
            return None
            
        # Double-check ownership at application layer (defense in depth)
        if not isolation.IsolationService.validate_resource_ownership(
            doc, self.user, self.collection_name
        ):
            return None
            
        return self.model_class(**core._strip_oid(doc))
    
    async def find_many(
        self,
        query: dict,
        station_id: Optional[str] = None,
        limit: int = 100,
        skip: int = 0
    ) -> list[T]:
        """List with automatic isolation filter"""
        isolated_query = isolation.IsolationService.apply_isolation_filter(
            query, self.user, self.collection_name, station_id
        )
        
        coll = await self._get_collection(station_id)
        cursor = coll.find(isolated_query, skip=skip, limit=limit).sort("created_at", -1)
        
        results = []
        async for doc in cursor:
            if isolation.IsolationService.validate_resource_ownership(
                doc, self.user, self.collection_name
            ):
                results.append(self.model_class(**core._strip_oid(doc)))
        return results
    
    async def create(self, data: dict, station_id: str) -> T:
        """Create with mandatory station assignment"""
        # Enforce: new resources must be assigned to a station the user owns
        if station_id not in self.user.get("assigned_stations", []):
            raise HTTPException(403, "Cannot create resource in unassigned station")
        
        # Inject isolation fields
        doc = {
            **data,
            "station_id": station_id,
            "created_by": self.user["id"],
            "created_at": core.now_iso(),
            "updated_at": core.now_iso()
        }
        
        coll = await self._get_collection(station_id)
        result = await coll.insert_one(doc)
        doc["_id"] = result.inserted_id
        return self.model_class(**core._strip_oid(doc))
```

### A. Rate Limiting Middleware with Station-Aware Throttling (`backend/middleware/rate_limit.py`)

```python
# === CREATE FILE: backend/middleware/rate_limit.py ===

import time
from fastapi import Request, HTTPException, status
from starlette.middleware.base import BaseHTTPMiddleware
from redis.asyncio import Redis  # For distributed rate limiting
from ..core import settings

class StationAwareRateLimit(BaseHTTPMiddleware):
    """
    Rate limiting with different thresholds per:
    - User role (founder vs staff)
    - Endpoint sensitivity (auth vs read vs write)
    - Station load (prevent abuse during peak hours)
    """
    
    def __init__(self, app, redis_url: str = None):
        super().__init__(app)
        self.redis = Redis.from_url(redis_url) if redis_url else None
        
        # Rate limit configurations per endpoint pattern
        self.limits = {
            # Authentication endpoints: strict limits to prevent brute force
            r"^/api/auth/.*": {"requests": 5, "window": 300, "key_prefix": "auth"},  # 5/5min
            
            # MPesa webhook: allow burst for callback reliability
            r"^/api/mpesa/webhook.*": {"requests": 100, "window": 60, "key_prefix": "mpesa"},  # 100/min
            
            # Sales creation: moderate limits per station to prevent flood
            r"^/api/sales.*": {"requests": 30, "window": 60, "key_prefix": "sales"},  # 30/min per station
            
            # Read endpoints: higher limits for reporting
            r"^/api/reports/.*": {"requests": 20, "window": 60, "key_prefix": "reports"},
            
            # Audit log queries: very strict (sensitive data)
            r"^/api/audit/.*": {"requests": 10, "window": 300, "key_prefix": "audit"},
            
            # Default fallback
            "default": {"requests": 60, "window": 60, "key_prefix": "generic"}
        }
    
    async def _get_rate_key(self, request: Request, limit_config: dict) -> str:
        """Generate Redis key for rate limit tracking"""
        prefix = limit_config["key_prefix"]
        
        # Extract user identifier
        user_id = "anon"
        auth_header = request.headers.get("authorization", "")
        if auth_header.startswith("Bearer "):
            # Decode JWT without full validation for rate limiting (lightweight)
            import jwt
            try:
                token = auth_header[7:]
                payload = jwt.decode(token, options={"verify_signature": False})
                user_id = payload.get("sub", "anon")
            except:
                pass
        
        # Add station context for station-scoped limits
        station_id = request.headers.get("x-station-id", "global")
        
        # Time bucket for sliding window
        window = limit_config["window"]
        bucket = int(time.time() // window)
        
        return f"ratelimit:{prefix}:{user_id}:{station_id}:{bucket}"
    
    async def _check_limit(self, key: str, limit_config: dict) -> bool:
        """Check and increment rate limit counter"""
        if not self.redis:
            # No Redis: skip rate limiting (development mode)
            return True
        
        max_requests = limit_config["requests"]
        window = limit_config["window"]
        
        # Use Redis pipeline for atomic increment + expiry
        pipe = self.redis.pipeline()
        pipe.incr(key)
        pipe.expire(key, window * 2)  # Keep for 2 windows to handle clock skew
        results = await pipe.execute()
        
        current_count = results[0]
        return current_count <= max_requests
    
    async def dispatch(self, request: Request, call_next):
        # Skip rate limiting for health checks and static assets
        if request.url.path in {"/health", "/metrics", "/docs", "/redoc", "/static"}:
            return await call_next(request)
        
        # Find matching limit config
        limit_config = self.limits.get("default")
        for pattern, config in self.limits.items():
            if pattern != "default" and request.url.path.startswith(pattern.replace(".*", "")):
                limit_config = config
                break
        
        # Generate rate limit key
        rate_key = await self._get_rate_key(request, limit_config)
        
        # Check limit
        if not await self._check_limit(rate_key, limit_config):
            # Log rate limit violation to audit trail (async, non-blocking)
            from ..services.audit import AuditService
            import asyncio
            asyncio.create_task(
                AuditService.log_access(
                    user_id="rate_limited",  # Special identifier
                    action="rate_limit_exceeded",
                    resource_type="api_endpoint",
                    resource_id=request.url.path,
                    ip_address=request.client.host if request.client else "unknown",
                    user_agent=request.headers.get("user-agent", "unknown"),
                    station_id=request.headers.get("x-station-id"),
                    reason=f"exceeded_{limit_config['requests']}_per_{limit_config['window']}s"
                )
            )
            
            raise HTTPException(
                status_code=status.HTTP_429_TOO_MANY_REQUESTS,
                detail="Rate limit exceeded. Please try again later.",
                headers={
                    "Retry-After": str(limit_config["window"]),
                    "X-RateLimit-Limit": str(limit_config["requests"]),
                    "X-RateLimit-Reset": str(int(time.time()) + limit_config["window"])
                }
            )
        
        # Proceed with request
        response = await call_next(request)
        
        # Add rate limit headers to response
        response.headers["X-RateLimit-Limit"] = str(limit_config["requests"])
        response.headers["X-RateLimit-Remaining"] = "N"  # Would require Redis GET, optional
        
        return response
```

### B. Request Validation & Sanitization Middleware (`backend/middleware/validation.py`)

```python
# === CREATE FILE: backend/middleware/validation.py ===

import re
from fastapi import Request, HTTPException, status
from starlette.middleware.base import BaseHTTPMiddleware
from pydantic import ValidationError

class RequestValidationMiddleware(BaseHTTPMiddleware):
    """
    Validate and sanitize all incoming requests:
    - Block SQL/NoSQL injection patterns
    - Validate Content-Type headers
    - Sanitize path/query parameters
    - Enforce JSON schema for mutation endpoints
    """
    
    # Patterns indicating injection attempts
    DANGEROUS_PATTERNS = [
        r"(\bunion\b.*\bselect\b)",  # SQL union injection
        r"(\bdrop\b.*\btable\b)",    # SQL drop table
        r"(\b\$where\b)",            # MongoDB $where injection
        r"(\beval\b\s*\()",          # JavaScript eval
        r"(\<script\b)",             # XSS script tag
        r"(\.\.\/)",                 # Path traversal
        r"(\b0x[0-9a-f]+\b)",        # Hex encoding bypass attempts
    ]
    
    COMPILED_PATTERNS = [re.compile(p, re.I) for p in DANGEROUS_PATTERNS]
    
    async def _scan_for_injection(self, text: str, context: str) -> bool:
        """Check text for dangerous patterns"""
        if not text:
            return False
        for pattern in self.COMPILED_PATTERNS:
            if pattern.search(text):
                # Log attempt to audit trail
                from ..services.audit import AuditService
                import asyncio
                asyncio.create_task(
                    AuditService.log_access(
                        user_id="injection_attempt",
                        action="security_block",
                        resource_type="request_validation",
                        resource_id=context,
                        ip_address="unknown",  # Will be filled by caller
                        reason=f"injection_pattern_detected:{pattern.pattern}"
                    )
                )
                return True
        return False
    
    async def dispatch(self, request: Request, call_next):
        # Only validate mutation endpoints and sensitive reads
        if request.method in {"GET", "HEAD", "OPTIONS"} and "/audit" not in request.url.path:
            return await call_next(request)
        
        # Validate Content-Type for POST/PUT/PATCH
        if request.method in {"POST", "PUT", "PATCH"}:
            content_type = request.headers.get("content-type", "")
            if "application/json" not in content_type and request.url.path.startswith("/api/"):
                raise HTTPException(
                    status_code=status.HTTP_415_UNSUPPORTED_MEDIA_TYPE,
                    detail="Content-Type must be application/json"
                )
        
        # Scan URL path and query params for injection
        if await self._scan_for_injection(request.url.path, f"path:{request.url.path}"):
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="Invalid request: suspicious pattern detected in URL"
            )
        
        for key, value in request.query_params.items():
            if await self._scan_for_injection(value, f"query:{key}"):
                raise HTTPException(
                    status_code=status.HTTP_400_BAD_REQUEST,
                    detail=f"Invalid request: suspicious pattern in parameter '{key}'"
                )
        
        # For JSON body requests, read and validate body
        if request.method in {"POST", "PUT", "PATCH"} and "application/json" in request.headers.get("content-type", ""):
            try:
                body = await request.json()
                
                # Recursively scan body values
                def scan_dict(d, path="body"):
                    if isinstance(d, dict):
                        for k, v in d.items():
                            scan_dict(v, f"{path}.{k}")
                    elif isinstance(d, list):
                        for i, v in enumerate(d):
                            scan_dict(v, f"{path}[{i}]")
                    elif isinstance(d, str) and await self._scan_for_injection(d, path):
                        raise ValueError(f"Injection pattern in {path}")
                
                scan_dict(body)
                
                # Re-attach validated body to request for downstream use
                # Note: In production, use a request body cache middleware
                request.state.validated_body = body
                
            except ValidationError as e:
                raise HTTPException(
                    status_code=status.HTTP_422_UNPROCESSABLE_ENTITY,
                    detail=f"Request validation failed: {str(e)}"
                )
            except ValueError as e:
                raise HTTPException(
                    status_code=status.HTTP_400_BAD_REQUEST,
                    detail=str(e)
                )
            except Exception as e:
                # Log parsing errors but don't leak details
                from ..services.audit import AuditService
                import asyncio
                asyncio.create_task(
                    AuditService.log_access(
                        user_id="parse_error",

                        resource_id=request.url.path,
                        ip_address=request.client.host if request.client else "unknown",
                        reason=f"body_parse_error:{type(e).__name__}"
                    )
                )
                raise HTTPException(
                    status_code=status.HTTP_400_BAD_REQUEST,
                    detail="Invalid request body format"
                )
        
        return await call_next(request)
```

```typescript
export async function secureFetch(
  url: string,
  options: SecureRequestOptions = {}
): Promise<Response> {
  const { stationId, skipPinValidation = false, ...fetchOptions } = options;
  
  // Get auth token from store
  const { user } = useAuthStore.getState();
  
  // Build headers
  const headers = new Headers(fetchOptions.headers);
  if (user?.token) {
    headers.set('Authorization', `Bearer ${user.token}`);
  }
  if (stationId) {
    headers.set('X-Station-ID', stationId);
  }
  headers.set('X-App-Version', import.meta.env.VITE_APP_VERSION || 'dev');
  headers.set('X-Device-Id', await getDeviceId());  // Implement device fingerprinting
  
  // Certificate pinning for production
  if (import.meta.env.PROD && !skipPinValidation) {
    const pins = CERTIFICATE_PINS.production;
    
    // CapacitorHttp native plugin supports pinning
    try {
      const response = await CapacitorHttp.request({
        url,
        method: fetchOptions.method || 'GET',
        headers: Object.fromEntries(headers.entries()),
        data: fetchOptions.body,
        connectTimeout: 15000,
        readTimeout: 30000,
        // Pinning configuration
        sslPinning: {
          mode: 'strict',  // Fail if no pin matches
          pins: pins,
          // Optional: include subdomains
          includeSubdomains: true,
        },
      });
      
      // Wrap native response in standard Response object
      return new Response(response.data, {
        status: response.status,
        statusText: response.statusText,
        headers: new Headers(response.headers),
      });
      
    } catch (error: any) {
      // Log pinning failures for security monitoring
      if (error.message?.includes('certificate') || error.message?.includes('pinning')) {
        console.error('Certificate pinning failed - possible MITM attack', {
          url,
          error: error.message,
          timestamp: new Date().toISOString(),
        });
        // Optionally report to security endpoint
        await reportSecurityEvent('cert_pinning_failure', { url, error: error.message });
      }
      throw error;
    }
  }
  
  // Fallback to standard fetch (dev/staging)
  return fetch(url, { ...fetchOptions, headers });
}

async function getDeviceId(): Promise<string> {
  // Generate stable device identifier without PII
  // Use Capacitor Device plugin with hashing
  const { Device } = await import('@capacitor/device');

  const encoder = new TextEncoder();

  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
}
```

```typescript
  async set(key: string, value: string, options?: {
    requireBiometric?: boolean;
    expiresIn?: number;  // seconds
  }): Promise<void> {
    const { requireBiometric = false, expiresIn } = options || {};
    
    // For values with expiry, wrap with metadata
    const payload = expiresIn 
      ? JSON.stringify({ value, expires: Date.now() + expiresIn * 1000 })
      : value;
    
    await SecureStoragePlugin.set({
      key,
      value: payload,
      useBiometric: requireBiometric && Platform.isNativePlatform,
    });
  },
  
  /**
   * Retrieve value with automatic expiry check and biometric prompt
   */
  async get(key: string, options?: {
    promptBiometric?: boolean;
    onBiometricFailure?: () => void;
  }): Promise<string | null> {
    const { promptBiometric = true, onBiometricFailure } = options || {};
    
    try {
      const result = await SecureStoragePlugin.get({
        key,
        useBiometric: promptBiometric && Platform.isNativePlatform,
      });
      
      // Parse expiry metadata if present
      try {
        const parsed = JSON.parse(result.value);
        if (parsed.expires && Date.now() > parsed.expires) {
          // Auto-delete expired value
          await SecureStoragePlugin.remove({ key });
          return null;
        }
        return parsed.value;
      } catch {
        // Not JSON-wrapped, return raw value
        return result.value;
      }
      
    } catch (error: any) {
      // Handle biometric failure gracefully
      if (error.message?.includes('biometric') || error.message?.includes('authentication')) {
        onBiometricFailure?.();
        return null;
      }
      // Key not found or other error
      return null;
    }
  },

   * Remove sensitive value

  },

        // Ignore errors during cleanup
      }
    }

export async function storeMpesaPin(pin: string): Promise<void> {
  // Never store raw PIN - hash with device-specific salt
  const deviceSalt = await getDeviceSpecificSalt();
  const encoder = new TextEncoder();
  const pinData = encoder.encode(pin + deviceSalt);
  const hashBuffer = await crypto.subtle.digest('SHA-256', pinData);
  const pinHash = Array.from(new Uint8Array(hashBuffer))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
  
  // Store hash with biometric requirement and 24h expiry

  });

/**
 * Helper: Verify MPesa PIN without exposing raw value
 */
export async function verifyMpesaPin(inputPin: string): Promise<boolean> {
  const storedHash = await SecureStorage.get('mpesa_pin_hash', { promptBiometric: true });
  if (!storedHash) return false;
  
  // Re-compute hash with same salt
  const deviceSalt = await getDeviceSpecificSalt();
  const encoder = new TextEncoder();
  const inputData = encoder.encode(inputPin + deviceSalt);
  const hashBuffer = await crypto.subtle.digest('SHA-256', inputData);
  const inputHash = Array.from(new Uint8Array(hashBuffer))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
  
  // Constant-time comparison to prevent timing attacks
  return storedHash.length === inputHash.length && 
         [...storedHash].every((char, i) => char === inputHash[i]);
}

  // Generate salt from device identifiers + app secret
  const { Device } = await import('@capacitor/device');
  const info = await Device.getInfo();

  const data = encoder.encode(`${info.uuid}-${info.model}-${appSecret}`);
  const hashBuffer = await crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(hashBuffer))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('')
    .substring(0, 32);  // Use first 32 chars as salt
```

### A. Secrets Loader with Fallback (`backend/core/secrets.py`)

```python
# === CREATE FILE: backend/core/secrets.py ===

import os
import json
import boto3
from functools import lru_cache
from typing import Optional, Dict, Any

class SecretsManager:
    """
    Unified secrets loader with fallback chain:
    1. AWS Secrets Manager (production)
    2. Environment variables (staging)
    3. Local .env file (development - never commit)
    
    Supports automatic rotation callbacks for database credentials.
    """
    
    def __init__(self, env: str = None):
        self.env = env or os.getenv("ENV", "development")
        self._secrets_cache: Dict[str, Any] = {}
        
        # Initialize AWS client only in production/staging
        if self.env in {"production", "staging"}:
            try:
                self.aws_client = boto3.client(
                    "secretsmanager",
                    region_name=os.getenv("AWS_REGION", "eu-central-1")
                )
            except Exception as e:
                print(f"⚠️  AWS Secrets Manager unavailable: {e}")
                self.aws_client = None
        else:
            self.aws_client = None
    
    @lru_cache(maxsize=128)
    def get(self, secret_name: str, field: str = None, default: Any = None) -> Any:
        """
        Retrieve secret with intelligent caching and fallback.
        
        Args:
            secret_name: Name of secret in AWS or env var prefix
            field: Optional field within JSON secret (e.g., "password")
            default: Fallback value if not found
        """
        # Check in-memory cache first
        cache_key = f"{secret_name}:{field}" if field else secret_name
        if cache_key in self._secrets_cache:
            return self._secrets_cache[cache_key]
        
        # Try AWS Secrets Manager (production/staging)
        if self.aws_client and self.env in {"production", "staging"}:
            try:
                response = self.aws_client.get_secret_value(SecretId=secret_name)
                secret_value = json.loads(response["SecretString"])
                
                # Extract field if specified
                value = secret_value.get(field) if field else secret_value
                
                # Cache successful retrieval
                self._secrets_cache[cache_key] = value
                return value
                
            except Exception as e:
                print(f"⚠️  Failed to fetch {secret_name} from AWS: {e}")
                # Fall through to env vars
        
        # Try environment variables
        env_key = secret_name.upper().replace("-", "_")
        if field:
            env_key = f"{env_key}_{field.upper()}"
        
        env_value = os.getenv(env_key)
        if env_value:
            # Attempt JSON parse for complex values
            try:
                parsed = json.loads(env_value)
                value = parsed.get(field) if field and isinstance(parsed, dict) else parsed
                self._secrets_cache[cache_key] = value
                return value
            except json.JSONDecodeError:
                self._secrets_cache[cache_key] = env_value
                return env_value
        
        # Try local .env file (development only)
        if self.env == "development":
            try:
                from dotenv import dotenv_values
                env_vars = dotenv_values(".env.local") or dotenv_values(".env")
                local_value = env_vars.get(env_key)
                if local_value:
                    self._secrets_cache[cache_key] = local_value
                    return local_value
            except ImportError:
                pass  # python-dotenv not installed
        
        # Return default or raise
        if default is not None:
            self._secrets_cache[cache_key] = default
            return default
        
        raise ValueError(
            f"Secret '{secret_name}' not found in any source. "
            f"Env: {self.env}, attempted keys: {cache_key}, {env_key}"
        )
    
    def invalidate_cache(self, secret_name: Optional[str] = None):
        """Clear cache for secret (use after rotation)"""
        if secret_name:
            # Invalidate all fields for this secret
            keys_to_remove = [k for k in self._secrets_cache if k.startswith(f"{secret_name}:")]
            for key in keys_to_remove:
```

```python
#!/usr/bin/env python3
"""
Automated secret rotation for SOC-2 compliance.
Rotates JWT secrets, audit HMAC keys, and database passwords.
Integrates with AWS Secrets Manager for production.
"""
import asyncio
import argparse
import hashlib
import secrets
from datetime import datetime, timedelta
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core.secrets import secrets, get_db_credentials
from backend.core import MONGO_URL, now_iso

async def rotate_jwt_secret(new_secret: str = None, grace_period_hours: int = 24):
    """
    Rotate JWT signing key with grace period for token validation.
    Old key remains valid for grace period to avoid user logout storms.
    """
    if not new_secret:
        new_secret = secrets.token_urlsafe(32)
    
    old_secret = secrets.get("fuelapp-jwt", "secret_key")
    
    # Store new secret with metadata
    rotation_record = {
        "id": f"jwt_rotation_{datetime.now().strftime('%Y%m%d_%H%M%S')}",
        "secret_name": "fuelapp-jwt",
        "field": "secret_key",
        "old_value_hash": hashlib.sha256(old_secret.encode()).hexdigest()[:16],
        "new_value_hash": hashlib.sha256(new_secret.encode()).hexdigest()[:16],
        "rotated_at": now_iso(),
        "rotated_by": "rotation_script_v1",
        "grace_period_ends": (datetime.now() + timedelta(hours=grace_period_hours)).isoformat(),
        "status": "active"
    }
    
    # In production: update AWS Secrets Manager
    if secrets.aws_client:
        # AWS Secrets Manager rotation requires Lambda integration
        # This script prepares the new value; actual rotation triggered by AWS
        print(f"🔄 Production: New JWT secret prepared. Trigger AWS rotation via console or Lambda.")
        print(f"   Secret ARN: {secrets.get('fuelapp-jwt', '_arn')}")
    else:
        # Development: update env var file
        print(f"🔄 Development: Update .env.local with new JWT_SECRET")
        print(f"   JWT_SECRET={new_secret}")
    
    # Log rotation to audit trail
    from backend.services.audit import AuditService
    await AuditService.log_access(
        user_id="system_rotation",
        action="secret_rotated",
        resource_type="jwt_secret",
        resource_id=rotation_record["id"],
        ip_address="127.0.0.1",
        user_agent="rotation-script/1.0",
        resource_snapshot_after={"grace_period_hours": grace_period_hours}
    )
    
    # Invalidate cache so new requests use new secret
    secrets.invalidate_cache("fuelapp-jwt")
    
    return rotation_record

async def rotate_audit_hmac_key(new_key: str = None):
    """Rotate HMAC key for audit log integrity"""
    if not new_key:
        new_key = secrets.token_urlsafe(32)
    
    if len(new_key) < 32:
        raise ValueError("New HMAC key must be at least 32 characters")
    
    old_key_hash = hashlib.sha256(
        secrets.get("fuelapp-audit", "hmac_key").encode()
    ).hexdigest()[:16]
    
    # Update secret store
    if secrets.aws_client:
        print(f"🔄 Production: Update AWS Secrets Manager for fuelapp-audit:hmac_key")
    else:
        print(f"🔄 Development: Update .env.local with new AUDIT_HMAC_KEY")
        print(f"   AUDIT_HMAC_KEY={new_key}")
    
    # Critical: Re-sign existing audit logs with new key? 
    # SOC-2: Keep old key for verifying historical logs
    # Store key versioning metadata
    key_version = {
        "version": datetime.now().strftime("%Y.%m"),
        "active_from": now_iso(),
        "previous_hash": old_key_hash,
        "current_hash": hashlib.sha256(new_key.encode()).hexdigest()[:16]
    }
    
    # Log rotation
    from backend.services.audit import AuditService
    await AuditService.log_access(
        user_id="system_rotation",
        action="secret_rotated",
        resource_type="audit_hmac_key",
        resource_id=key_version["version"],
        ip_address="127.0.0.1",
        user_agent="rotation-script/1.0",
        resource_snapshot_after=key_version
    )
    
    secrets.invalidate_cache("fuelapp-audit")
    
    # Update in-memory reference for AuditService
    from backend.services import audit
    audit.AuditService._AUDIT_HMAC_KEY = new_key
    
    return key_version

async def rotate_database_password(new_password: str = None):
    """
    Rotate MongoDB user password with zero-downtime connection update.
    Requires MongoDB user with password change privileges.
    """
    # Generate strong password if not provided
    if not new_password:
        new_password = secrets.token_urlsafe(24)
    
    # Get current credentials
    creds = get_db_credentials()
    
    # Update MongoDB user password
    client = AsyncIOMotorClient(
        f"mongodb://{creds['username']}:{creds['password']}@{creds['host']}:{creds['port']}",
        replicaSet=creds.get("replica_set")
    )
    
    try:
        admin_db = client.admin
        await admin_db.command(
            "updateUser", creds["username"],
            pwd=new_password
        )
        print(f"✅ MongoDB password rotated for user: {creds['username']}")
        
        # Update secret store
        if secrets.aws_client:
            # Prepare AWS Secrets Manager update
            print(f"🔄 Production: Update fuelapp-mongodb:password in AWS Secrets Manager")
        else:
            print(f"🔄 Development: Update .env.local with new DB password")
        
        # Log rotation
        from backend.services.audit import AuditService
        await AuditService.log_access(
            user_id="system_rotation",
            action="secret_rotated",
            resource_type="database_password",
            resource_id=creds["username"],
            ip_address="127.0.0.1",
            user_agent="rotation-script/1.0"
        )
        
        # Invalidate cache
        secrets.invalidate_cache("fuelapp-mongodb")
        
        return {"status": "success", "username": creds["username"]}
        
    finally:
        await client.close()

async def main():
    parser = argparse.ArgumentParser(description="Rotate secrets for SOC-2 compliance")
    parser.add_argument("--secret", required=True, choices=["jwt", "audit_hmac", "db_password"],
                       help="Which secret to rotate")
    parser.add_argument("--new-value", help="New secret value (auto-generated if omitted)")
    parser.add_argument("--grace-period", type=int, default=24,
                       help="Grace period in hours for JWT rotation (default: 24)")
    parser.add_argument("--dry-run", action="store_true",
                       help="Show what would be done without making changes")
    
    args = parser.parse_args()
    
    if args.dry_run:
        print(f"🔍 DRY RUN: Would rotate {args.secret}")
        if args.secret == "jwt":
            print(f"   Grace period: {args.grace_period} hours")
        return
    
    print(f"🔄 Starting rotation: {args.secret}")
    
    if args.secret == "jwt":
        result = await rotate_jwt_secret(args.new_value, args.grace_period)
    elif args.secret == "audit_hmac":
        result = await rotate_audit_hmac_key(args.new_value)
    elif args.secret == "db_password":
        result = await rotate_database_password(args.new_value)
    
    print(f"✅ Rotation complete: {result}")
    print(f"📋 Next steps:")
    print(f"   1. Deploy updated secrets to all application instances")
    print(f"   2. Monitor logs for authentication failures during grace period")
    print(f"   3. Archive old key material per retention policy")

if __name__ == "__main__":
    asyncio.run(main())
```

```python
            if params:
                separator = "&" if "?" in base else "?"
                base += f"{separator}{('&'.join(params))}"
        return base
```

### A. Audit Log Backup Script (`scripts/backup_audit_logs.py`)

```python
#!/usr/bin/env python3
"""
SOC-2 compliant backup of audit logs to immutable S3 storage.
Features:
- Incremental backups using MongoDB change streams
- Cross-region replication for disaster recovery
- Integrity verification with HMAC chaining
- Automated retention enforcement
"""
import asyncio
import hashlib
import hmac
import json
import os
import boto3
from datetime import datetime, timedelta
from motor.motor_asyncio import AsyncIOMotorClient, ChangeStream
from backend.core import MONGO_URL, AUDIT_COLLECTION, secrets

class AuditLogBackup:
    def __init__(self, backup_bucket: str, region: str = "eu-central-1"):
        self.backup_bucket = backup_bucket
        self.region = region
        self.s3 = boto3.client("s3", region_name=region)
        self.glacier = boto3.client("glacier", region_name=region)
        
        # HMAC key for backup integrity (separate from live audit key)
        self.backup_hmac_key = secrets.get("fuelapp-audit", "backup_hmac_key")
    
    async def incremental_backup(self, start_after: str = None):
        """
        Perform incremental backup using MongoDB change streams.
        Only backs up new/modified audit logs since last backup.
        """
        client = AsyncIOMotorClient(MONGO_URL)
        db = client.get_database()
        audit_coll = db[AUDIT_COLLECTION]
        
        # Build backup filename with timestamp
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_key = f"audit-backups/incremental/{timestamp}.jsonl"
        
        # Track backup metadata
        backup_metadata = {
            "backup_id": f"audit_{timestamp}",
            "start_time": datetime.now().isoformat(),
            "source_collection": AUDIT_COLLECTION,
            "region": self.region,
            "record_count": 0,
            "integrity": {}
        }
        
        # Open S3 multipart upload for large backups
        upload = self.s3.create_multipart_upload(
            Bucket=self.backup_bucket,
            Key=backup_key,
            ContentType="application/jsonl",
            Metadata={
                "backup-type": "incremental",
                "compliance": "SOC2",
                "immutable": "true"
            }
        )
        
        parts = []
        part_number = 1
        buffer = []
        buffer_size = 0
        MIN_PART_SIZE = 5 * 1024 * 1024  # 5MB minimum for multipart
        
        prev_log_hash = None
        
        # Query for new logs since last backup marker
        query = {"timestamp": {"$gt": start_after}} if start_after else {}
        cursor = audit_coll.find(query).sort("timestamp", 1)
        
        async for log in cursor:
            # Add chain hash for tamper evidence
            critical = {k: log[k] for k in ["user_id", "action", "resource_type", "timestamp"]}
            log_hash = hashlib.sha256(json.dumps(critical, sort_keys=True).encode()).hexdigest()
            
            # Chain to previous log in this backup
            if prev_log_hash:
                log["backup_chain_hash"] = hashlib.sha256(
                    (prev_log_hash + log_hash).encode()
                ).hexdigest()
            prev_log_hash = log_hash
            
            # Sign with backup HMAC key
            message = json.dumps({
                "log_id": log.get("id"),
                "timestamp": log["timestamp"],
                "log_hash": log_hash
            }, sort_keys=True)
            log["backup_signature"] = hmac.new(
                self.backup_hmac_key.encode(),
                message.encode(),
                hashlib.sha256
            ).hexdigest()
            
            # Add to buffer
            buffer.append(json.dumps(log) + "\n")
            buffer_size += len(buffer[-1])
            backup_metadata["record_count"] += 1
            
            # Flush buffer when part size reached
            if buffer_size >= MIN_PART_SIZE:
                part_data = "".join(buffer).encode()
                part = self.s3.upload_part(
                    Bucket=self.backup_bucket,
                    Key=backup_key,
                    PartNumber=part_number,
                    UploadId=upload["UploadId"],
                    Body=part_data,
                    ContentMD5=self._compute_md5(part_data)
                )
                parts.append({"PartNumber": part_number, "ETag": part["ETag"]})
                part_number += 1
                buffer = []
                buffer_size = 0
        
        # Upload remaining buffer
        if buffer:
            part_data = "".join(buffer).encode()
            part = self.s3.upload_part(
                Bucket=self.backup_bucket,
                Key=backup_key,
                PartNumber=part_number,
                UploadId=upload["UploadId"],
                Body=part_data,
                ContentMD5=self._compute_md5(part_data)
            )
            parts.append({"PartNumber": part_number, "ETag": part["ETag"]})
        
        # Complete multipart upload
        self.s3.complete_multipart_upload(
            Bucket=self.backup_bucket,
            Key=backup_key,
            UploadId=upload["UploadId"],
            MultipartUpload={"Parts": parts}
        )
        
        # Apply Object Lock for immutability (GOVERNANCE mode)
        self.s3.put_object_retention(
            Bucket=self.backup_bucket,
            Key=backup_key,
            Retention={
                "Mode": "GOVERNANCE",
                "RetainUntilDate": (datetime.now() + timedelta(days=2555)).date()
            }
        )
        
        # Enable cross-region replication to DR region
        if os.getenv("DR_REGION"):
            self._enable_cross_region_replication(backup_key)
        
        # Update backup metadata with integrity info
        backup_metadata["end_time"] = datetime.now().isoformat()
        backup_metadata["integrity"]["final_chain_hash"] = prev_log_hash
        backup_metadata["integrity"]["s3_etag"] = parts[-1]["ETag"] if parts else None
        
        # Store metadata separately for cataloging
        metadata_key = backup_key.replace(".jsonl", ".metadata.json")
        self.s3.put_object(
            Bucket=self.backup_bucket,
            Key=metadata_key,
            Body=json.dumps(backup_metadata, indent=2),
            ContentType="application/json"
        )
        
        await client.close()
        
        print(f"✅ Backup complete: {backup_key}")
        print(f"   Records: {backup_metadata['record_count']}")
        print(f"   Chain hash: {prev_log_hash[:16]}...")
        
        return backup_metadata
    
    def _compute_md5(self, data: bytes) -> str:
        """Compute base64-encoded MD5 for S3 Content-MD5 header"""
        import base64
        return base64.b64encode(hashlib.md5(data).digest()).decode()
    
    def _enable_cross_region_replication(self, key: str):
        """Configure cross-region replication for disaster recovery"""
        dr_region = os.getenv("DR_REGION", "us-east-1")
        
        # Copy to DR region with same Object Lock settings
        self.s3.copy_object(
            CopySource={"Bucket": self.backup_bucket, "Key": key},
            Bucket=self.backup_bucket,
            Key=key.replace("eu-central-1", dr_region),  # Path-based region separation
            MetadataDirective="REPLACE",
            Metadata={
                "backup-type": "incremental",
                "compliance": "SOC2",
                "immutable": "true",
                "replicated-from": self.region
            }
        )
        
        # Apply retention in DR region
        dr_s3 = boto3.client("s3", region_name=dr_region)
        dr_s3.put_object_retention(
            Bucket=self.backup_bucket,
            Key=key.replace("eu-central-1", dr_region),
            Retention={
                "Mode": "GOVERNANCE",
                "RetainUntilDate": (datetime.now() + timedelta(days=2555)).date()
            }
        )
    
    async def verify_backup_integrity(self, backup_key: str) -> dict:
        """Verify backup hasn't been tampered with"""
        # Download backup metadata
        metadata_obj = self.s3.get_object(
            Bucket=self.backup_bucket,
            Key=backup_key.replace(".jsonl", ".metadata.json")
        )
        metadata = json.loads(metadata_obj["Body"].read())
        
        # Stream backup and verify chain
        backup_obj = self.s3.get_object(
            Bucket=self.backup_bucket,
            Key=backup_key
        )
        
        prev_hash = None
        verified_count = 0
        
        for line in backup_obj["Body"].iter_lines():
            if not line or line.startswith(b"#"):
                continue
                
            log = json.loads(line)
            
            # Verify chain hash
            critical = {k: log[k] for k in ["user_id", "action", "resource_type", "timestamp"]}
            expected_hash = hashlib.sha256(
                json.dumps(critical, sort_keys=True).encode()
            ).hexdigest()
            
            if log.get("log_hash") != expected_hash:
                return {"valid": False, "error": f"Hash mismatch at record {verified_count}"}
            
            # Verify chain
            if prev_hash and log.get("backup_chain_hash"):
                expected_chain = hashlib.sha256(
                    (prev_hash + expected_hash).encode()
                ).hexdigest()
                if log.get("backup_chain_hash") != expected_chain:
                    return {"valid": False, "error": f"Chain broken at record {verified_count}"}
            
            # Verify signature
            message = json.dumps({
                "log_id": log.get("id"),
                "timestamp": log["timestamp"],
                "log_hash": expected_hash
            }, sort_keys=True)
            expected_sig = hmac.new(
                self.backup_hmac_key.encode(),
                message.encode(),
                hashlib.sha256
            ).hexdigest()
            if log.get("backup_signature") != expected_sig:
                return {"valid": False, "error": f"Signature mismatch at record {verified_count}"}
            
            prev_hash = expected_hash
            verified_count += 1
        
        # Verify final chain hash matches metadata
        if metadata["integrity"]["final_chain_hash"] != prev_hash:
            return {"valid": False, "error": "Final chain hash mismatch"}
        
        return {
            "valid": True,
            "records_verified": verified_count,
            "backup_id": metadata["backup_id"]
        }

async def main():
    backup = AuditLogBackup(
        backup_bucket=os.getenv("AUDIT_BACKUP_BUCKET", "fuelapp-audit-backups"),
        region=os.getenv("AWS_REGION", "eu-central-1")
    )
    
    # Get last backup timestamp from catalog
    last_backup = await _get_last_backup_timestamp()
    
    # Perform incremental backup
    result = await backup.incremental_backup(start_after=last_backup)

    # Update catalog with successful backup

    print(f"✅ Backup verified and cataloged")
    return 0
```

```python
    def __init__(self, app, metrics_port: int = 9090):
        self.app = app
        # Start Prometheus metrics server
        start_http_server(metrics_port)
    
    async def __call__(self, scope, receive, send):
        if scope["type"] != "http":
            return await self.app(scope, receive, send)
        
        start_time = time.time()
        
        # Wrap send to capture response status
        async def wrapped_send(message):
            if message["type"] == "http.response.start":
                # Record auth attempt metrics
                if scope["path"].startswith("/api/auth/"):
                    status = "success" if message["status"] < 400 else "failure"
                    SECURITY_METRICS["auth_attempts_total"].labels(
                        status=status,
                        method=scope["method"],
                        user_role="unknown"  # Would extract from auth middleware
                    ).inc()
                
                # Record permission denial metrics
                if message["status"] == 403:
                    # Extract context from scope (would be set by RBAC middleware)
                    SECURITY_METRICS["permission_denials_total"].labels(
                        action=scope.get("rbac_action", "unknown"),
                        data_type=scope.get("rbac_data_type", "unknown"),
                        user_role=scope.get("user_role", "unknown"),
                        station_id=scope.get("station_id", "unknown")
                    ).inc()
                
                # Record rate limit metrics
                if message["status"] == 429:
                    SECURITY_METRICS["rate_limit_exceeded_total"].labels(
                        endpoint=scope["path"],
                        user_id=scope.get("user_id", "anon"),
                        station_id=scope.get("station_id", "global")
                    ).inc()
            
            await send(message)
        
        try:
            return await self.app(scope, receive, wrapped_send)
        finally:
            # Record request duration for security-relevant endpoints
            if any(prefix in scope["path"] for prefix in ["/api/auth", "/api/audit", "/api/sales"]):
                duration = time.time() - start_time
                # Could add to histogram if needed
                pass
```

### A. Migration: Add `station_id` to Existing Records (`scripts/migrate_add_station_id.py`)

```python
#!/usr/bin/env python3
"""
Migration script: Backfill station_id on existing records for users with single-station assignment.
Run during maintenance window. Idempotent and reversible.
"""
import asyncio
import sys
from datetime import datetime
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, new_id, now_iso

# Collections requiring station_id isolation
TARGET_COLLECTIONS = ["sales", "inventory", "expenses", "invoices", "employees"]

async def migrate_collection(db, collection_name: str, dry_run: bool = True):
    """Add station_id to documents missing it, based on user assignment"""
    coll = db[collection_name]
    
    # Find docs without station_id but with created_by reference
    query = {"station_id": {"$exists": False}, "created_by": {"$exists": True}}
    cursor = coll.find(query, {"created_by": 1, "_id": 1})
    
    updated = 0
    errors = 0
    
    async for doc in cursor:
        try:
            # Fetch user to get their assigned stations
            user = await db.users.find_one(
                {"id": doc["created_by"]}, 
                {"assigned_stations": 1, "role": 1}
            )
            
            if not user or not user.get("assigned_stations"):
                # Cannot determine station - log for manual review
                print(f"⚠️  {collection_name}:{doc['_id']} - No station assignment for user {doc['created_by']}")
                errors += 1
                continue
            
            # Use first assigned station (assumes single-station users for legacy data)
            station_id = user["assigned_stations"][0]
            
            if dry_run:
                print(f"🔍 WOULD UPDATE {collection_name}:{doc['_id']} → station_id={station_id}")
                updated += 1
            else:
                result = await coll.update_one(
                    {"_id": doc["_id"]},
                    {"$set": {
                        "station_id": station_id,
                        "migrated_at": now_iso(),
                        "migrated_by": "migration_script_v1"
                    }}
                )
                if result.modified_count > 0:
                    updated += 1
                    # Log migration to audit trail
                    await db.audit_logs_soc2.insert_one({
                        "id": new_id(),
                        "user_id": "system_migration",
                        "user_email": "migration@fuelapp.internal",
                        "user_role": "system",
                        "action": "data_migration",
                        "resource_type": collection_name,
                        "resource_id": str(doc["_id"]),
                        "station_id": station_id,
                        "ip_address": "127.0.0.1",
                        "user_agent": "migration-script/1.0",
                        "timestamp": now_iso(),
                        "resource_snapshot_after": {"station_id_added": station_id},
                        "log_hash": "",  # Will be computed by audit service in production
                        "log_signature": ""
                    })
                    
        except Exception as e:
            print(f"❌ Error migrating {collection_name}:{doc.get('_id')}: {e}")
            errors += 1
    
    return updated, errors

async def main():
    if len(sys.argv) > 1 and sys.argv[1] == "--execute":
        dry_run = False
        print("🚀 EXECUTING MIGRATION (not dry-run)")
    else:
        dry_run = True
        print("🔍 DRY RUN MODE - No changes will be made")
        print("   Use '--execute' flag to apply changes")
    
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()  # Uses env DB_NAME
    
    total_updated = 0
    total_errors = 0
    
    for coll_name in TARGET_COLLECTIONS:
        print(f"\n📦 Processing collection: {coll_name}")
        updated, errors = await migrate_collection(db, coll_name, dry_run=dry_run)
        total_updated += updated
        total_errors += errors
        print(f"   ✓ Updated: {updated} | Errors: {errors}")
    
    print(f"\n{'='*60}")
    print(f"SUMMARY: {total_updated} records processed, {total_errors} errors")
    if dry_run:
        print("⚠️  This was a DRY RUN. Run with '--execute' to apply changes.")
    print(f"{'='*60}")
    
    await client.close()
    return 0 if total_errors == 0 else 1

if __name__ == "__main__":
    exit_code = asyncio.run(main())
    sys.exit(exit_code)
```

### B. Migration: Initialize Permissions for Existing Users (`scripts/migrate_init_permissions.py`)

```python
#!/usr/bin/env python3
"""
Migration: Generate RBAC permissions for existing users based on role/tier/station assignments.
Creates permission documents in new 'permissions' collection for fast lookup.
"""
import asyncio
import sys
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, new_id, now_iso
from backend.services.rbac import RBACService  # Reuse permission generation logic

async def migrate_user_permissions(db, user: dict, dry_run: bool = True):
    """Generate and store permissions for a single user"""
    user_id = user.get("id")
    if not user_id:
        return 0
    
    # Generate permissions using existing RBAC logic
    permissions = await RBACService._load_user_permissions(user_id)
    
    if dry_run:
        print(f"🔍 WOULD CREATE {len(permissions)} permissions for user {user_id} ({user.get('email')})")
        return len(permissions)
    
    # Store in dedicated permissions collection for O(1) lookup
    perm_doc = {
        "id": new_id(),
        "user_id": user_id,
        "permissions": [p.model_dump() for p in permissions],
        "generated_at": now_iso(),
        "generated_by": "migration_script_v1",
        "role_snapshot": user.get("role"),
        "tier_snapshot": user.get("tier"),
        "stations_snapshot": user.get("assigned_stations")
    }
    
    # Upsert: replace existing if re-running migration
    await db.permissions.update_one(
        {"user_id": user_id},
        {"$set": perm_doc},
        upsert=True
    )
    
    # Log to audit trail
    await db.audit_logs_soc2.insert_one({
        "id": new_id(),
        "user_id": "system_migration",
        "user_email": "migration@fuelapp.internal", 
        "user_role": "system",
        "action": "permissions_generated",
        "resource_type": "user_permissions",
        "resource_id": user_id,
        "station_id": user.get("assigned_stations", [None])[0] if user.get("assigned_stations") else None,
        "ip_address": "127.0.0.1",
        "user_agent": "migration-script/1.0",
        "timestamp": now_iso(),
        "resource_snapshot_after": {"permission_count": len(permissions)},
        "log_hash": "",
        "log_signature": ""
    })
    
    return len(permissions)

async def main():
    dry_run = not (len(sys.argv) > 1 and sys.argv[1] == "--execute")
    
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()
    
    # Process users in batches to avoid memory issues
    batch_size = 100
    skip = 0
    total_users = 0
    total_perms = 0
    
    while True:
        users = await db.users.find(
            {"permissions": {"$exists": False}},  # Only migrate users without pre-computed permissions
            {"id": 1, "email": 1, "role": 1, "tier": 1, "assigned_stations": 1}
        ).skip(skip).limit(batch_size).to_list(batch_size)
        
        if not users:
            break
            
        for user in users:
            perms = await migrate_user_permissions(db, user, dry_run=dry_run)
            total_perms += perms
            total_users += 1
        
        skip += batch_size
        print(f"   Processed {total_users} users...")
    
    print(f"\n{'='*60}")
    print(f"MIGRATION COMPLETE: {total_users} users, {total_perms} permissions generated")
    if dry_run:
        print("⚠️  DRY RUN - Use '--execute' to apply changes")
    print(f"{'='*60}")
    
    await client.close()
    return 0

if __name__ == "__main__":
    exit(asyncio.run(main()))
```

### C. Rollback Script (`scripts/rollback_migration.py`)

```python
#!/usr/bin/env python3
"""
Emergency rollback: Remove migration-added fields if issues occur.
USE WITH CAUTION - Only run if migration caused data corruption.
"""
import asyncio
import sys
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL

async def rollback_collection(db, collection_name: str):
    """Remove migration-added fields from a collection"""
    coll = db[collection_name]
    
    # Remove station_id if it was added by migration (has migrated_by marker)
    result = await coll.update_many(
        {"migrated_by": "migration_script_v1"},
        {"$unset": {
            "station_id": "",
            "migrated_at": "",
            "migrated_by": ""
        }}
    )
    return result.modified_count

async def rollback_permissions(db):
    """Remove migration-generated permission documents"""
    result = await db.permissions.delete_many({
        "generated_by": "migration_script_v1"
    })
    return result.deleted_count

async def main():
    if len(sys.argv) > 1 and sys.argv[1] == "--confirm":
        print("⚠️  CONFIRMING ROLLBACK - This will REMOVE migration changes!")
        confirmation = input("Type 'ROLLBACK' to proceed: ")
        if confirmation != "ROLLBACK":
            print("❌ Rollback cancelled")
            return 1
    else:
        print("🔒 Rollback requires '--confirm' flag and manual confirmation")
        return 1
```

```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import type { Permission, ActionType, DataType, TeamScope } from '../../backend/core';

export interface UserSession {
  id: string;
  email: string;
  role: 'founder' | 'owner' | 'manager' | 'staff' | 'auditor' | 'guest';
  tier: 'free' | 'pro' | 'enterprise';
  assigned_stations: string[];
  permissions: Permission[];
  token: string;
  expires_at: string;
}

interface AuthState {
  user: UserSession | null;
  activeStationId: string | null;
  
  // Actions
  login: (credentials: {email: string, password: string}) => Promise<void>;
  logout: () => void;
  setActiveStation: (stationId: string | null) => void;
  
  // Permission helpers
  can: (action: ActionType, dataType: DataType, stationId?: string) => boolean;
  hasStationAccess: (stationId: string) => boolean;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      activeStationId: null,
      
      login: async (credentials) => {
        const response = await fetch('/api/auth/login', {
          method: 'POST',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify(credentials)
        });
        
        if (!response.ok) throw new Error('Authentication failed');
        
        const session = await response.json();
        set({ user: session });
        
        // Auto-select first station if none active
        if (!get().activeStationId && session.assigned_stations?.[0]) {
          set({ activeStationId: session.assigned_stations[0] });
        }
      },
      
      logout: () => {
        // Revoke token server-side
        fetch('/api/auth/logout', {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${get().user?.token}`
          }
        }).catch(() => {}); // Ignore errors
        
        set({ user: null, activeStationId: null });
      },
      
      setActiveStation: (stationId) => {
        const { user } = get();
        // Validate station is in user's assignments
        if (stationId && !user?.assigned_stations.includes(stationId)) {
          console.warn(`Station ${stationId} not assigned to user`);
          return;
        }
        set({ activeStationId: stationId });
      },
      
      // Permission evaluation helper for UI conditional rendering
      can: (action: ActionType, dataType: DataType, stationId?: string) => {
        const { user, activeStationId } = get();
        if (!user) return false;
        
        // Founder bypass
        if (user.role === 'founder') return true;
        
        // Guest restrictions
        if (user.role === 'guest') {
          return action === 'read' && ['station', 'inventory'].includes(dataType);
        }
        
        // Check permissions array (pre-computed by backend)
        const targetStation = stationId || activeStationId;
        
        return user.permissions.some(perm => {
          if (perm.action !== action || perm.data_type !== dataType) return false;
          
          switch (perm.team_scope) {
            case 'global':
              return ['pro', 'enterprise'].includes(user.tier);
            case 'station':
              return perm.station_id === targetStation && 
                     user.assigned_stations.includes(targetStation || '');
            case 'team':
              // Implement team logic if used
              return false;
            case 'personal':
              // For personal scope, UI should filter by owner_id client-side
              return true;
            default:
              return false;
          }
        });
      },
      
      hasStationAccess: (stationId: string) => {
        const { user } = get();
        return user?.assigned_stations.includes(stationId) || user?.role === 'founder';
      }
    }),
    {
      name: 'fuelapp-auth',
      partialize: (state) => ({ 
        user: state.user, 
        activeStationId: state.activeStationId 
      }),
      // Re-validate session on app resume
      onRehydrateStorage: () => (state) => {
        if (state?.user?.expires_at) {
          const expires = new Date(state.user.expires_at);
          if (expires < new Date()) {
            console.log('Session expired, clearing auth');
            state.logout();
          }
        }
      }
    }
  )
);
```

```typescript
import { createTRPCProxyClient, httpLink, TRPCLink } from '@trpc/client';
import { observable } from '@trpc/server/observable';
import type { AppRouter } from '../../backend/routers/_app';
import { useAuthStore } from '../stores/authStore';

// Custom link to inject auth & station headers on every request
const authLink: TRPCLink<AppRouter> = () => {
  return ({ next, op }) => {
    return observable((observer) => {
      const { user, activeStationId } = useAuthStore.getState();
      
      const headers = new Headers();
      
      if (user?.token) {
        headers.set('Authorization', `Bearer ${user.token}`);
      }
      
      // Critical: Always send active station for isolation enforcement
      if (activeStationId) {
        headers.set('X-Station-ID', activeStationId);
      }
      
      // Add request metadata for audit logging
      headers.set('X-Request-Source', 'mobile-app');
      headers.set('X-App-Version', import.meta.env.VITE_APP_VERSION || 'dev');
      
      // Continue with modified context
      const unsubscribe = next({
        ...op,
        context: {
          ...op.context,
          headers: Object.fromEntries(headers.entries())
        }
      }).subscribe(observer);
      
      return unsubscribe;
    });
  };
};

export const trpc = createTRPCProxyClient<AppRouter>({
  links: [
    authLink,  // Must be first to modify headers
    httpLink({
      url: import.meta.env.VITE_API_URL || '/api',
    }),
  ],
});

// Type-safe hook for station-scoped queries
export function useStationQuery<T>(
  queryKey: string[],
  queryFn: (stationId: string) => Promise<T>,
  options?: { enabled?: boolean; staleTime?: number }
) {
  const { activeStationId } = useAuthStore();
  
  return {
    ...useQuery({
      queryKey: [...queryKey, activeStationId],
      queryFn: () => activeStationId ? queryFn(activeStationId) : Promise.resolve(null),
      enabled: options?.enabled !== false && !!activeStationId,
      staleTime: options?.staleTime || 30000, // 30s default
      // Auto-refetch when station changes
      refetchOnMount: 'always',
    }),
    // Helper to manually trigger with specific station (for cross-station views)
    fetchForStation: (stationId: string) => queryFn(stationId)
  };
}
```

```tsx
import { ReactNode } from 'react';
import { useAuthStore } from '../stores/authStore';
import type { ActionType, DataType } from '../../backend/core';

interface PermissionGateProps {
  action: ActionType;
  dataType: DataType;
  stationId?: string;  // Optional: check for specific station
  fallback?: ReactNode;  // What to show if denied
  children: ReactNode;
}

export function PermissionGate({
  action,
  dataType,
  stationId,
  fallback = null,
  children
}: PermissionGateProps) {
  const { can, user } = useAuthStore();
  
  // Show fallback for unauthenticated users
  if (!user) return <>{fallback}</>;
  
  // Evaluate permission
  const hasPermission = can(action, dataType, stationId);
  
  if (!hasPermission) {
    // Log denied access attempt for audit (client-side telemetry)
    console.debug(`Permission denied: ${action} on ${dataType}`, {
      userId: user.id,
      stationId: stationId || useAuthStore.getState().activeStationId,
      role: user.role
    });
    
    return <>{fallback}</>;
  }
  
  return <>{children}</>;
}
```

### A. Audit Log Export Script for SOC-2 Auditors (`scripts/export_audit_soc2.py`)

```python
#!/usr/bin/env python3
"""
Export audit logs in SOC-2 compliant format for external auditors.
Generates immutable, signed CSV/JSONL with chain-of-custody metadata.
"""
import asyncio
import csv
import hashlib
import hmac
import json
import sys
from datetime import datetime, timedelta
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, AUDIT_COLLECTION

# SOC-2 required fields per CC7.2 (Audit & Monitoring)
SOC2_REQUIRED_FIELDS = [
    "timestamp", "user_id", "user_email", "action", "resource_type", 
    "resource_id", "ip_address", "station_id", "log_hash", "log_signature"
]

async def export_audit_logs(
    start_date: str,
    end_date: str,
    output_format: str = "jsonl",
    output_path: str = "audit_export",
    signer_key: str = None
):
    """Export logs with SOC-2 compliance metadata"""
    
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()
    
    # Query with time range
    query = {
        "timestamp": {
            "$gte": start_date,
            "$lte": end_date
        }
    }
    
    # Project only required fields + integrity fields
    projection = {field: 1 for field in SOC2_REQUIRED_FIELDS}
    projection["_id"] = 0  # Exclude MongoDB internal ID
    
    cursor = db[AUDIT_COLLECTION].find(query, projection).sort("timestamp", 1)
    
    # Prepare export metadata (required for chain of custody)
    export_metadata = {
        "export_id": hashlib.sha256(f"{start_date}{end_date}{datetime.now().isoformat()}".encode()).hexdigest()[:16],
        "export_timestamp": datetime.now().isoformat(),
        "date_range": {"start": start_date, "end": end_date},
        "field_schema": SOC2_REQUIRED_FIELDS,
        "record_count": 0,
        "integrity": {}
    }
    
    records = []
    prev_hash = None
    
    async for doc in cursor:
        # Re-compute hash for verification
        critical_fields = {k: doc[k] for k in ["user_id", "action", "resource_type", "timestamp", "ip_address"]}
        computed_hash = hashlib.sha256(
            json.dumps(critical_fields, sort_keys=True).encode()
        ).hexdigest()
        
        # Add chain hash
        doc["chain_hash"] = hashlib.sha256(
            (prev_hash or "") + computed_hash
        ).hexdigest()
        prev_hash = doc["chain_hash"]
        
        records.append(doc)
        export_metadata["record_count"] += 1
    
    # Sign the export manifest
    if signer_key:
        manifest_json = json.dumps(export_metadata, sort_keys=True)
        export_metadata["manifest_signature"] = hmac.new(
            signer_key.encode(),
            manifest_json.encode(),
            hashlib.sha256
        ).hexdigest()
    
    # Write output
    if output_format == "jsonl":
        with open(f"{output_path}.jsonl", "w") as f:
            # Write metadata as first line (prefixed)
            f.write(f"# METADATA: {json.dumps(export_metadata)}\n")
            for record in records:
                f.write(json.dumps(record) + "\n")
    
    elif output_format == "csv":
        with open(f"{output_path}.csv", "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=SOC2_REQUIRED_FIELDS + ["chain_hash"])
            writer.writeheader()
            # Write metadata as comment
            f.write(f"# SOC-2 Export Metadata: {json.dumps(export_metadata)}\n")
            for record in records:
                writer.writerow(record)
    
    await client.close()
```

```python
#!/usr/bin/env python3
"""
Generate quarterly access review report for SOC-2 compliance.
Lists all users with permissions, station assignments, and last activity.
"""
import asyncio
import csv
from datetime import datetime, timedelta
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, now_iso

async def generate_access_report(output_path: str = "access_review.csv"):
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()
    
    # Fetch all non-system users with permissions
    users_cursor = db.users.find(
        {"role": {"$nin": ["system", "migration"]}},
        {
            "id": 1, "email": 1, "role": 1, "tier": 1,
            "assigned_stations": 1, "permissions": 1,
            "last_login": 1, "created_at": 1
        }
    )
    
    # Fetch recent audit activity for each user (last 90 days)
    cutoff = (datetime.now() - timedelta(days=90)).isoformat()
    
    with open(output_path, "w", newline="") as f:
        writer = csv.writer(f)
        # SOC-2 required headers for access reviews
        writer.writerow([
            "User ID", "Email", "Role", "Tier", "Assigned Stations",
            "Permission Count", "Last Login", "Account Created",
            "Recent Actions (90d)", "Review Status", "Reviewer Notes"
        ])
        
        async for user in users_cursor:
            # Count recent activity from audit logs
            activity_count = await db.audit_logs_soc2.count_documents({
                "user_id": user["id"],
                "timestamp": {"$gte": cutoff}
            })
            
            # Format stations list
            stations = ", ".join(user.get("assigned_stations", [])) or "NONE"
            
            # Permission count (from cached array or compute)
            perm_count = len(user.get("permissions", []))
            
            writer.writerow([
                user["id"],
                user["email"],
                user["role"],
                user["tier"],
                stations,
                perm_count,
                user.get("last_login", "NEVER"),
                user.get("created_at", "UNKNOWN"),
                f"{activity_count} events",
                "PENDING_REVIEW",  # Default status
                ""  # Notes for reviewer
            ])
    
    await client.close()
    
    # Generate summary metadata
    summary = {
        "report_id": f"AR-{datetime.now().strftime('%YQ%m')}",
        "generated_at": now_iso(),
        "review_period": "90 days",
        "total_users": await db.users.count_documents({"role": {"$nin": ["system", "migration"]}}),
        "output_file": output_path,
        "next_review_due": (datetime.now() + timedelta(days=90)).isoformat()
    }
    
    # Write summary alongside CSV
    with open(output_path.replace(".csv", ".metadata.json"), "w") as f:
        import json
        json.dump(summary, f, indent=2)
    
    print(f"✅ Access review report generated: {output_path}")
    print(f"📋 Summary: {summary['total_users']} users, review due {summary['next_review_due']}")
    return summary

if __name__ == "__main__":
    asyncio.run(generate_access_report())
```

# 2️⃣ Backend: FastAPI Google Sign-In Integration (`backend/routers/auth.py`)

```python
# === MODIFY: backend/routers/auth.py ===

from fastapi import APIRouter, HTTPException, Depends, Request, Header
from google.oauth2 import id_token
from google.auth.transport import requests as google_requests
from ..core import settings, db, new_id, now_iso
from ..services.rbac import RBACService
from ..services.audit import AuditService
import jwt
from datetime import datetime, timedelta

router = APIRouter(prefix="/api/auth", tags=["authentication"])

# Google OAuth configuration
GOOGLE_CLIENT_IDS = {
    "web": settings.GOOGLE_CLIENT_ID_WEB,
    "android": settings.GOOGLE_CLIENT_ID_ANDROID, 
    "ios": settings.GOOGLE_CLIENT_ID_IOS,
}

class GoogleTokenPayload(BaseModel):
    """Validated Google ID token payload"""
    sub: str  # Google user ID
    email: str
    email_verified: bool
    name: str
    picture: str
    aud: str  # Must match one of our client IDs
    exp: int  # Expiration timestamp
    iat: int  # Issued at timestamp

async def verify_google_token(id_token_str: str, platform: str = "web") -> GoogleTokenPayload:
    """
    Verify Google ID token signature and claims.
    Raises HTTPException if verification fails.
    """
    client_id = GOOGLE_CLIENT_IDS.get(platform)
    if not client_id:
        raise HTTPException(400, f"Unsupported platform: {platform}")
    
    try:
        # Verify token with Google's public keys
        # This checks: signature, expiration, audience, issuer
        token_info = id_token.verify_oauth2_token(
            id_token_str,
            google_requests.Request(),
            audience=client_id,
            clock_skew_in_seconds=10  # Allow 10s clock drift
        )
        
        # Validate required claims
        if not token_info.get("email"):
            raise ValueError("Token missing email claim")
        if not token_info.get("email_verified"):
            raise ValueError("Email not verified by Google")
            
        return GoogleTokenPayload(**token_info)
        
    except ValueError as e:
        # Token invalid, expired, or wrong audience
        raise HTTPException(
            status_code=401,
            detail=f"Invalid Google token: {str(e)}",
            headers={"WWW-Authenticate": "Bearer error=invalid_token"}
        )
    except Exception as e:
        # Network error, Google API down, etc.
        raise HTTPException(
            status_code=503,
            detail=f"Google authentication service unavailable: {str(e)}"
        )

@router.post("/google")
async def google_signin(
    request: Request,
    token_data: dict,  # {id_token: "...", platform: "web|android|ios"}
    x_station_id: Optional[str] = Header(None)
):
    """
    Handle Google Sign-In: verify token, create/find user, return app JWT.
    
    Request body:
    {
      "id_token": "<Google ID token from client>",
      "platform": "web|android|ios",
      "station_id": "optional-station-id"  # For auto-assignment
    }
    """
    # Validate request
    id_token_str = token_data.get("id_token")
    platform = token_data.get("platform", "web")
    
    if not id_token_str:
        raise HTTPException(400, "Missing id_token")
    
    # Verify Google token
    google_payload = await verify_google_token(id_token_str, platform)
    
    # Audit: Log authentication attempt
    await AuditService.log_access(
        user_id=google_payload.sub,  # Google ID as external identifier
        user_email=google_payload.email,
        user_role="google_user",
        action="google_signin_attempt",
        resource_type="authentication",
        ip_address=request.client.host if request.client else "unknown",
        user_agent=request.headers.get("user-agent", "unknown"),
        station_id=x_station_id,
        resource_snapshot_after={"platform": platform, "email_verified": google_payload.email_verified}
    )
    
    # Find or create user in our system
    user = await db.users.find_one({"email": google_payload.email})
    
    if not user:
        # New user: auto-register with default role
        user = {
            "id": new_id(),
            "email": google_payload.email,
            "name": google_payload.name,
            "avatar": google_payload.picture,
            "google_id": google_payload.sub,
            "email_verified": True,  # Google already verified
            "role": "staff",  # Default role for Google signups
            "tier": "free",
            "assigned_stations": [],
            "permissions": [],
            "created_at": now_iso(),
            "last_login": now_iso(),
            "auth_providers": ["google"],
            "status": "active"
        }
        await db.users.insert_one(user)
        
        # Audit: New user registration
        await AuditService.log_access(
            user_id=user["id"],
            user_email=user["email"],
            user_role=user["role"],
            action="user_registered",
            resource_type="user",
            resource_id=user["id"],
            ip_address=request.client.host if request.client else "unknown",
            resource_snapshot_after={"auth_provider": "google", "auto_assigned_role": "staff"}
        )
    else:
        # Existing user: update profile and last login
        update_ops = {
            "$set": {
                "name": google_payload.name,
                "avatar": google_payload.picture,
                "google_id": google_payload.sub,
                "last_login": now_iso(),
                "email_verified": True
            },
            "$addToSet": {"auth_providers": "google"}  # Ensure google is in list
        }
        await db.users.update_one({"_id": user["_id"]}, update_ops)
        user = await db.users.find_one({"email": google_payload.email})
    
    # Auto-assign station if requested and user has permission
    if x_station_id and x_station_id not in user.get("assigned_stations", []):
        # Check if user can be assigned to this station (founder/owner only)
        # For Google signups, require manual station assignment by admin
        # Log attempt for audit
        await AuditService.log_access(
            user_id=user["id"],
            action="station_assignment_attempt",
            resource_type="station",
            resource_id=x_station_id,
            reason="google_signin_auto_assign_rejected",
            ip_address=request.client.host if request.client else "unknown"
        )
    
    # Generate our app's JWT token (not Google's)
    app_token = create_app_jwt_token(user)
    
    # Audit: Successful authentication
    await AuditService.log_access(
        user_id=user["id"],
        user_email=user["email"],
        user_role=user["role"],
        action="google_signin_success",
        resource_type="authentication",
        ip_address=request.client.host if request.client else "unknown",
        resource_snapshot_after={"token_issued": True, "station_count": len(user.get("assigned_stations", []))}
    )
    
    # Return user session (exclude sensitive fields)
    return {
        "access_token": app_token,
        "token_type": "bearer",
        "expires_in": 86400,  # 24 hours
        "user": {
            "id": user["id"],
            "email": user["email"],
            "name": user["name"],
            "avatar": user["avatar"],
            "role": user["role"],
            "tier": user["tier"],
            "assigned_stations": user.get("assigned_stations", []),
            "permissions": user.get("permissions", []),
            "email_verified": user["email_verified"]
        }
    }

def create_app_jwt_token(user: dict) -> str:
    """Generate our application's JWT token for authenticated sessions"""
    payload = {
        "sub": user["id"],
        "email": user["email"],
        "role": user["role"],
        "tier": user["tier"],
        "stations": user.get("assigned_stations", []),
```

```typescript
export async function initGoogleAuth(): Promise<void> {

/**
 * Trigger Google Sign-In flow
 * Returns Google user data including idToken for backend verification
 */
export async function signInWithGoogle(stationId?: string): Promise<GoogleUser> {
  const platform = Capacitor.getPlatform();
  
  try {
    if (Capacitor.isNativePlatform()) {
      // === NATIVE MOBILE (Android/iOS) ===
      const result = await GoogleAuth.signIn();
      
      if (!result?.authentication?.idToken) {
        throw new Error('Google Sign-In returned no ID token');
      }
      
      return {
        id: result.userId,
        email: result.email,
        name: result.name,
        photoUrl: result.imageUrl,
        idToken: result.authentication.idToken, // 🔑 Critical for backend verification
        accessToken: result.authentication.accessToken,
      };
      
    } else {
      // === WEB BROWSER ===
      // Use Google Identity Services (GIS) library
      return new Promise((resolve, reject) => {
        // @ts-ignore - google.accounts is loaded via script tag
        const client = google.accounts.oauth2.initTokenClient({
          client_id: GOOGLE_CONFIG.web.clientId,
          scope: GOOGLE_CONFIG.web.scope,
          prompt: GOOGLE_CONFIG.web.prompt,
          callback: (response: any) => {
            if (response.error) {
              reject(new Error(`Google OAuth error: ${response.error}`));
              return;
            }
            
            // Fetch user profile using access token
            fetch('https://www.googleapis.com/oauth2/v3/userinfo', {
              headers: { Authorization: `Bearer ${response.access_token}` }
            })
            .then(res => res.json())
            .then(profile => {
              resolve({
                id: profile.sub,
                email: profile.email,
                name: profile.name,
                photoUrl: profile.picture,
                idToken: response.id_token, // 🔑 Critical: GIS returns id_token in response
                accessToken: response.access_token,
              });
            })
            .catch(reject);
          },
        });
        
        client.requestAccessToken();
      });
    }
    
  } catch (error: any) {
    // Handle common errors
    if (error.message?.includes('10') || error.message?.includes('DEVELOPER_ERROR')) {
      console.error('❌ Android SHA-1 not registered in Google Console');
      console.error('Fix: Run `./gradlew signingReport` and add SHA-1 to Google Cloud Console');
    }
    if (error.message?.includes('redirect_uri_mismatch')) {
      console.error('❌ Redirect URI mismatch');
      console.error('Fix: Add exact URI to Google Cloud Console → Credentials → Authorized redirect URIs');
    }
    
    // Log error for debugging
    console.error('Google Sign-In failed:', error);
    throw error;
  }
}
```

# 🔐 Security Best Practices for Google Sign-In

1. **Never trust client-side claims** - Always verify `id_token` on backend with `google.oauth2.id_token.verify_oauth2_token` [[5]]
2. **Use short-lived app tokens** - Issue your own JWT with 24h expiry, not Google's token
3. **Validate email_verified** - Only allow sign-in if `email_verified: true` from Google
4. **Audit all auth events** - Log successful/failed attempts to `audit_logs_soc2`
5. **Rate limit auth endpoints** - Prevent brute force on `/api/auth/google`
6. **Rotate Google client secrets** - Quarterly via Google Console; update env vars
7. **Use platform-specific client IDs** - Never reuse web client ID for mobile apps

### B. Replace `backend/routers/auth.py` Google Endpoint — Direct Token Verification

```python
# === REPLACE /auth/google endpoint in backend/routers/auth.py ===

from google.oauth2 import id_token
from google.auth.transport import requests as google_requests

class GoogleAuthBody(BaseModel):
    id_token: str = Field(..., min_length=100, description="Google ID token from client")
    platform: Literal["web", "android", "ios"] = "web"

@router.post("/auth/google", response_model=TokenResponse)
async def google_auth_exchange(body: GoogleAuthBody, request: Request):
    """
    Verify Google ID token directly and create FuelPro session.
    SOC-2 compliant: token verified server-side, no proxy dependency.
    """
    # Validate platform client ID exists
    client_id = GOOGLE_CLIENT_IDS.get(body.platform)
    if not client_id or not is_google_client_id_valid(client_id):
        log.error("Google client ID misconfigured for platform: %s", body.platform)
        raise HTTPException(
            status_code=500,
            detail="Authentication service misconfigured",
            headers={"X-Error-Code": "GOOGLE_CONFIG_INVALID"}
        )
    
    try:
        # Verify token signature, expiration, audience, issuer with Google's public keys
        token_info = id_token.verify_oauth2_token(
            body.id_token,
            google_requests.Request(),
            audience=client_id,
            clock_skew_in_seconds=10  # Allow minor clock drift
        )
        
        # Validate required claims
        if not token_info.get("email") or not token_info.get("email_verified"):
            raise ValueError("Token missing verified email")
            
    except ValueError as e:
        # Log detailed error for debugging (never expose to client)
```

```typescript
import { Capacitor } from '@capacitor/core';
import { GoogleAuth } from '@codetrix-studio/capacitor-google-auth';

const GOOGLE_CONFIG = {
  web: { clientId: import.meta.env.VITE_GOOGLE_CLIENT_ID_WEB },
  android: { clientId: import.meta.env.VITE_GOOGLE_CLIENT_ID_ANDROID },
  ios: { clientId: import.meta.env.VITE_GOOGLE_CLIENT_ID_IOS },
};

export async function signInWithGoogle(platform: 'web' | 'android' | 'ios' = 'web') {
  const config = GOOGLE_CONFIG[platform];
  
  if (!config.clientId) {
    throw new Error(`Google client ID not configured for ${platform}`);
  }

  if (Capacitor.isNativePlatform()) {
    // Native mobile: use Capacitor plugin
    await GoogleAuth.initialize({
      clientId: config.clientId,
      scopes: ['email', 'profile'],
      grantOfflineAccess: false,
    });
    
    const result = await GoogleAuth.signIn();
    
    if (!result?.authentication?.idToken) {
      throw new Error('Google Sign-In returned no ID token');
    }
    
    return {
      idToken: result.authentication.idToken,
      email: result.email,
      name: result.name,
      photoUrl: result.imageUrl,
    };
  } else {
    // Web: use Google Identity Services
    return new Promise((resolve, reject) => {
      // @ts-ignore - google.accounts loaded via script tag in index.html
      const client = google.accounts.oauth2.initTokenClient({
        client_id: config.clientId,
        scope: 'email profile',
        prompt: 'select_account',
        callback: (response: any) => {
          if (response.error) {
            reject(new Error(`Google OAuth: ${response.error}`));
            return;
          }
          // Fetch profile + return ID token
          fetch('https://www.googleapis.com/oauth2/v3/userinfo', {
            headers: { Authorization: `Bearer ${response.access_token}` }
          })
          .then(res => res.json())
          .then(profile => resolve({
            idToken: response.id_token,
            email: profile.email,
            name: profile.name,
            photoUrl: profile.picture,
          }))
          .catch(reject);
        },
      });
      client.requestAccessToken();
    });
  }
}
```
