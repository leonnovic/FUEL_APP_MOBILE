# 4️⃣ DATA MIGRATION SCRIPTS
## Backfill Existing Data with Station Isolation & Audit History

### A. Migration: Add `station_id` to Existing Records (`scripts/migrate_add_station_id.py`)

```python
#!/usr/bin/env python3
"""
Migration script: Backfill station_id on existing records for users with single-station assignment.
Run during maintenance window. Idempotent and reversible.
"""
import asyncio
import sys
from datetime import datetime
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, new_id, now_iso

# Collections requiring station_id isolation
TARGET_COLLECTIONS = ["sales", "inventory", "expenses", "invoices", "employees"]

async def migrate_collection(db, collection_name: str, dry_run: bool = True):
    """Add station_id to documents missing it, based on user assignment"""
    coll = db[collection_name]
    
    # Find docs without station_id but with created_by reference
    query = {"station_id": {"$exists": False}, "created_by": {"$exists": True}}
    cursor = coll.find(query, {"created_by": 1, "_id": 1})
    
    updated = 0
    errors = 0
    
    async for doc in cursor:
        try:
            # Fetch user to get their assigned stations
            user = await db.users.find_one(
                {"id": doc["created_by"]}, 
                {"assigned_stations": 1, "role": 1}
            )
            
            if not user or not user.get("assigned_stations"):
                # Cannot determine station - log for manual review
                print(f"⚠️  {collection_name}:{doc['_id']} - No station assignment for user {doc['created_by']}")
                errors += 1
                continue
            
            # Use first assigned station (assumes single-station users for legacy data)
            station_id = user["assigned_stations"][0]
            
            if dry_run:
                print(f"🔍 WOULD UPDATE {collection_name}:{doc['_id']} → station_id={station_id}")
                updated += 1
            else:
                result = await coll.update_one(
                    {"_id": doc["_id"]},
                    {"$set": {
                        "station_id": station_id,
                        "migrated_at": now_iso(),
                        "migrated_by": "migration_script_v1"
                    }}
                )
                if result.modified_count > 0:
                    updated += 1
                    # Log migration to audit trail
                    await db.audit_logs_soc2.insert_one({
                        "id": new_id(),
                        "user_id": "system_migration",
                        "user_email": "migration@fuelapp.internal",
                        "user_role": "system",
                        "action": "data_migration",
                        "resource_type": collection_name,
                        "resource_id": str(doc["_id"]),
                        "station_id": station_id,
                        "ip_address": "127.0.0.1",
                        "user_agent": "migration-script/1.0",
                        "timestamp": now_iso(),
                        "resource_snapshot_after": {"station_id_added": station_id},
                        "log_hash": "",  # Will be computed by audit service in production
                        "log_signature": ""
                    })
                    
        except Exception as e:
            print(f"❌ Error migrating {collection_name}:{doc.get('_id')}: {e}")
            errors += 1
    
    return updated, errors

async def main():
    if len(sys.argv) > 1 and sys.argv[1] == "--execute":
        dry_run = False
        print("🚀 EXECUTING MIGRATION (not dry-run)")
    else:
        dry_run = True
        print("🔍 DRY RUN MODE - No changes will be made")
        print("   Use '--execute' flag to apply changes")
    
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()  # Uses env DB_NAME
    
    total_updated = 0
    total_errors = 0
    
    for coll_name in TARGET_COLLECTIONS:
        print(f"\n📦 Processing collection: {coll_name}")
        updated, errors = await migrate_collection(db, coll_name, dry_run=dry_run)
        total_updated += updated
        total_errors += errors
        print(f"   ✓ Updated: {updated} | Errors: {errors}")
    
    print(f"\n{'='*60}")
    print(f"SUMMARY: {total_updated} records processed, {total_errors} errors")
    if dry_run:
        print("⚠️  This was a DRY RUN. Run with '--execute' to apply changes.")
    print(f"{'='*60}")
    
    await client.close()
    return 0 if total_errors == 0 else 1

if __name__ == "__main__":
    exit_code = asyncio.run(main())
    sys.exit(exit_code)
```

### B. Migration: Initialize Permissions for Existing Users (`scripts/migrate_init_permissions.py`)

```python
#!/usr/bin/env python3
"""
Migration: Generate RBAC permissions for existing users based on role/tier/station assignments.
Creates permission documents in new 'permissions' collection for fast lookup.
"""
import asyncio
import sys
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, new_id, now_iso
from backend.services.rbac import RBACService  # Reuse permission generation logic

async def migrate_user_permissions(db, user: dict, dry_run: bool = True):
    """Generate and store permissions for a single user"""
    user_id = user.get("id")
    if not user_id:
        return 0
    
    # Generate permissions using existing RBAC logic
    permissions = await RBACService._load_user_permissions(user_id)
    
    if dry_run:
        print(f"🔍 WOULD CREATE {len(permissions)} permissions for user {user_id} ({user.get('email')})")
        return len(permissions)
    
    # Store in dedicated permissions collection for O(1) lookup
    perm_doc = {
        "id": new_id(),
        "user_id": user_id,
        "permissions": [p.model_dump() for p in permissions],
        "generated_at": now_iso(),
        "generated_by": "migration_script_v1",
        "role_snapshot": user.get("role"),
        "tier_snapshot": user.get("tier"),
        "stations_snapshot": user.get("assigned_stations")
    }
    
    # Upsert: replace existing if re-running migration
    await db.permissions.update_one(
        {"user_id": user_id},
        {"$set": perm_doc},
        upsert=True
    )
    
    # Log to audit trail
    await db.audit_logs_soc2.insert_one({
        "id": new_id(),
        "user_id": "system_migration",
        "user_email": "migration@fuelapp.internal", 
        "user_role": "system",
        "action": "permissions_generated",
        "resource_type": "user_permissions",
        "resource_id": user_id,
        "station_id": user.get("assigned_stations", [None])[0] if user.get("assigned_stations") else None,
        "ip_address": "127.0.0.1",
        "user_agent": "migration-script/1.0",
        "timestamp": now_iso(),
        "resource_snapshot_after": {"permission_count": len(permissions)},
        "log_hash": "",
        "log_signature": ""
    })
    
    return len(permissions)

async def main():
    dry_run = not (len(sys.argv) > 1 and sys.argv[1] == "--execute")
    
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()
    
    # Process users in batches to avoid memory issues
    batch_size = 100
    skip = 0
    total_users = 0
    total_perms = 0
    
    while True:
        users = await db.users.find(
            {"permissions": {"$exists": False}},  # Only migrate users without pre-computed permissions
            {"id": 1, "email": 1, "role": 1, "tier": 1, "assigned_stations": 1}
        ).skip(skip).limit(batch_size).to_list(batch_size)
        
        if not users:
            break
            
        for user in users:
            perms = await migrate_user_permissions(db, user, dry_run=dry_run)
            total_perms += perms
            total_users += 1
        
        skip += batch_size
        print(f"   Processed {total_users} users...")
    
    print(f"\n{'='*60}")
    print(f"MIGRATION COMPLETE: {total_users} users, {total_perms} permissions generated")
    if dry_run:
        print("⚠️  DRY RUN - Use '--execute' to apply changes")
    print(f"{'='*60}")
    
    await client.close()
    return 0

if __name__ == "__main__":
    exit(asyncio.run(main()))
```

### C. Rollback Script (`scripts/rollback_migration.py`)

```python
#!/usr/bin/env python3
"""
Emergency rollback: Remove migration-added fields if issues occur.
USE WITH CAUTION - Only run if migration caused data corruption.
"""
import asyncio
import sys
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL

async def rollback_collection(db, collection_name: str):
    """Remove migration-added fields from a collection"""
    coll = db[collection_name]
    
    # Remove station_id if it was added by migration (has migrated_by marker)
    result = await coll.update_many(
        {"migrated_by": "migration_script_v1"},
        {"$unset": {
            "station_id": "",
            "migrated_at": "",
            "migrated_by": ""
        }}
    )
    return result.modified_count

async def rollback_permissions(db):
    """Remove migration-generated permission documents"""
    result = await db.permissions.delete_many({
        "generated_by": "migration_script_v1"
    })
    return result.deleted_count

async def main():
    if len(sys.argv) > 1 and sys.argv[1] == "--confirm":
        print("⚠️  CONFIRMING ROLLBACK - This will REMOVE migration changes!")
        confirmation = input("Type 'ROLLBACK' to proceed: ")
        if confirmation != "ROLLBACK":
            print("❌ Rollback cancelled")
            return 1
    else:
        print("🔒 Rollback requires '--confirm' flag and manual confirmation")
        return 1
    
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()
    
    print("🔄 Rolling back collection migrations...")
    for coll_name in ["sales", "inventory", "expenses", "invoices", "employees"]:
        count = await rollback_collection(db, coll_name)
        print(f"   {coll_name}: removed {count} migrated fields")
    
    print("🔄 Rolling back permissions...")
    perm_count = await rollback_permissions(db)
    print(f"   permissions: removed {perm_count} documents")
    
    print("✅ Rollback complete")
    await client.close()
    return 0

if __name__ == "__main__":
    exit(asyncio.run(main()))
```

### A. Auth Store with Permission Cache (`frontend/src/stores/authStore.ts`)

```typescript
// === CREATE FILE: frontend/src/stores/authStore.ts ===
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import type { Permission, ActionType, DataType, TeamScope } from '../../backend/core';

export interface UserSession {
  id: string;
  email: string;
  role: 'founder' | 'owner' | 'manager' | 'staff' | 'auditor' | 'guest';
  tier: 'free' | 'pro' | 'enterprise';
  assigned_stations: string[];
  permissions: Permission[];
  token: string;
  expires_at: string;
}

interface AuthState {
  user: UserSession | null;
  activeStationId: string | null;
  
  // Actions
  login: (credentials: {email: string, password: string}) => Promise<void>;
  logout: () => void;
  setActiveStation: (stationId: string | null) => void;
  
  // Permission helpers
  can: (action: ActionType, dataType: DataType, stationId?: string) => boolean;
  hasStationAccess: (stationId: string) => boolean;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      activeStationId: null,
      
      login: async (credentials) => {
        const response = await fetch('/api/auth/login', {
          method: 'POST',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify(credentials)
        });
        
        if (!response.ok) throw new Error('Authentication failed');
        
        const session = await response.json();
        set({ user: session });
        
        // Auto-select first station if none active
        if (!get().activeStationId && session.assigned_stations?.[0]) {
          set({ activeStationId: session.assigned_stations[0] });
        }
      },
      
      logout: () => {
        // Revoke token server-side
        fetch('/api/auth/logout', {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${get().user?.token}`
          }
        }).catch(() => {}); // Ignore errors
        
        set({ user: null, activeStationId: null });
      },
      
      setActiveStation: (stationId) => {
        const { user } = get();
        // Validate station is in user's assignments
        if (stationId && !user?.assigned_stations.includes(stationId)) {
          console.warn(`Station ${stationId} not assigned to user`);
          return;
        }
        set({ activeStationId: stationId });
      },
      
      // Permission evaluation helper for UI conditional rendering
      can: (action: ActionType, dataType: DataType, stationId?: string) => {
        const { user, activeStationId } = get();
        if (!user) return false;
        
        // Founder bypass
        if (user.role === 'founder') return true;
        
        // Guest restrictions
        if (user.role === 'guest') {
          return action === 'read' && ['station', 'inventory'].includes(dataType);
        }
        
        // Check permissions array (pre-computed by backend)
        const targetStation = stationId || activeStationId;
        
        return user.permissions.some(perm => {
          if (perm.action !== action || perm.data_type !== dataType) return false;
          
          switch (perm.team_scope) {
            case 'global':
              return ['pro', 'enterprise'].includes(user.tier);
            case 'station':
              return perm.station_id === targetStation && 
                     user.assigned_stations.includes(targetStation || '');
            case 'team':
              // Implement team logic if used
              return false;
            case 'personal':
              // For personal scope, UI should filter by owner_id client-side
              return true;
            default:
              return false;
          }
        });
      },
      
      hasStationAccess: (stationId: string) => {
        const { user } = get();
        return user?.assigned_stations.includes(stationId) || user?.role === 'founder';
      }
    }),
    {
      name: 'fuelapp-auth',
      partialize: (state) => ({ 
        user: state.user, 
        activeStationId: state.activeStationId 
      }),
      // Re-validate session on app resume
      onRehydrateStorage: () => (state) => {
        if (state?.user?.expires_at) {
          const expires = new Date(state.user.expires_at);
          if (expires < new Date()) {
            console.log('Session expired, clearing auth');
            state.logout();
          }
        }
      }
    }
  )
);
```

```typescript
    return observable((observer) => {
      const { user, activeStationId } = useAuthStore.getState();
      
      const headers = new Headers();
      
      if (user?.token) {
        headers.set('Authorization', `Bearer ${user.token}`);
      }
      
      // Critical: Always send active station for isolation enforcement
      if (activeStationId) {
        headers.set('X-Station-ID', activeStationId);
      }
      
      // Add request metadata for audit logging
      headers.set('X-Request-Source', 'mobile-app');
      headers.set('X-App-Version', import.meta.env.VITE_APP_VERSION || 'dev');
      
      // Continue with modified context
      const unsubscribe = next({
        ...op,
        context: {
          ...op.context,
          headers: Object.fromEntries(headers.entries())
        }
      }).subscribe(observer);
      
      return unsubscribe;
    });
  };
};

export const trpc = createTRPCProxyClient<AppRouter>({
  links: [
    authLink,  // Must be first to modify headers
    httpLink({
      url: import.meta.env.VITE_API_URL || '/api',
    }),
  ],
});

// Type-safe hook for station-scoped queries
export function useStationQuery<T>(
  queryKey: string[],
  queryFn: (stationId: string) => Promise<T>,
  options?: { enabled?: boolean; staleTime?: number }
) {
  const { activeStationId } = useAuthStore();
  
  return {
    ...useQuery({
      queryKey: [...queryKey, activeStationId],
      queryFn: () => activeStationId ? queryFn(activeStationId) : Promise.resolve(null),
      enabled: options?.enabled !== false && !!activeStationId,
      staleTime: options?.staleTime || 30000, // 30s default
      // Auto-refetch when station changes
      refetchOnMount: 'always',
    }),
    // Helper to manually trigger with specific station (for cross-station views)
    fetchForStation: (stationId: string) => queryFn(stationId)
  };
}
```

```tsx
// === CREATE FILE: frontend/src/components/PermissionGate.tsx ===
import { ReactNode } from 'react';
import { useAuthStore } from '../stores/authStore';
import type { ActionType, DataType } from '../../backend/core';

interface PermissionGateProps {
  action: ActionType;
  dataType: DataType;
  stationId?: string;  // Optional: check for specific station
  fallback?: ReactNode;  // What to show if denied
  children: ReactNode;
}

export function PermissionGate({
  action,
  dataType,
  stationId,
  fallback = null,
  children
}: PermissionGateProps) {
  const { can, user } = useAuthStore();
  
  // Show fallback for unauthenticated users
  if (!user) return <>{fallback}</>;
  
  // Evaluate permission
  const hasPermission = can(action, dataType, stationId);
  
  if (!hasPermission) {
    // Log denied access attempt for audit (client-side telemetry)
    console.debug(`Permission denied: ${action} on ${dataType}`, {
      userId: user.id,
      stationId: stationId || useAuthStore.getState().activeStationId,
      role: user.role
    });
    
    return <>{fallback}</>;
  }
  
  return <>{children}</>;
}
```

### A. Audit Log Export Script for SOC-2 Auditors (`scripts/export_audit_soc2.py`)

```python
#!/usr/bin/env python3
"""
Export audit logs in SOC-2 compliant format for external auditors.
Generates immutable, signed CSV/JSONL with chain-of-custody metadata.
"""
import asyncio
import csv
import hashlib
import hmac
import json
import sys
from datetime import datetime, timedelta
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, AUDIT_COLLECTION

# SOC-2 required fields per CC7.2 (Audit & Monitoring)
SOC2_REQUIRED_FIELDS = [
    "timestamp", "user_id", "user_email", "action", "resource_type", 
    "resource_id", "ip_address", "station_id", "log_hash", "log_signature"
]

async def export_audit_logs(
    start_date: str,
    end_date: str,
    output_format: str = "jsonl",
    output_path: str = "audit_export",
    signer_key: str = None
):
    """Export logs with SOC-2 compliance metadata"""
    
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()
    
    # Query with time range
    query = {
        "timestamp": {
            "$gte": start_date,
            "$lte": end_date
        }
    }
    
    # Project only required fields + integrity fields
    projection = {field: 1 for field in SOC2_REQUIRED_FIELDS}
    projection["_id"] = 0  # Exclude MongoDB internal ID
    
    cursor = db[AUDIT_COLLECTION].find(query, projection).sort("timestamp", 1)
    
    # Prepare export metadata (required for chain of custody)
    export_metadata = {
        "export_id": hashlib.sha256(f"{start_date}{end_date}{datetime.now().isoformat()}".encode()).hexdigest()[:16],
        "export_timestamp": datetime.now().isoformat(),
        "date_range": {"start": start_date, "end": end_date},
        "field_schema": SOC2_REQUIRED_FIELDS,
        "record_count": 0,
        "integrity": {}
    }
    
    records = []
    prev_hash = None
    
    async for doc in cursor:
        # Re-compute hash for verification
        critical_fields = {k: doc[k] for k in ["user_id", "action", "resource_type", "timestamp", "ip_address"]}
        computed_hash = hashlib.sha256(
            json.dumps(critical_fields, sort_keys=True).encode()
        ).hexdigest()
        
        # Add chain hash
        doc["chain_hash"] = hashlib.sha256(
            (prev_hash or "") + computed_hash
        ).hexdigest()
        prev_hash = doc["chain_hash"]
        
        records.append(doc)
        export_metadata["record_count"] += 1
    
    # Sign the export manifest
    if signer_key:
        manifest_json = json.dumps(export_metadata, sort_keys=True)
        export_metadata["manifest_signature"] = hmac.new(
            signer_key.encode(),
            manifest_json.encode(),
            hashlib.sha256
        ).hexdigest()
    
    # Write output
    if output_format == "jsonl":
        with open(f"{output_path}.jsonl", "w") as f:
            # Write metadata as first line (prefixed)
            f.write(f"# METADATA: {json.dumps(export_metadata)}\n")
            for record in records:
                f.write(json.dumps(record) + "\n")
    
    elif output_format == "csv":
        with open(f"{output_path}.csv", "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=SOC2_REQUIRED_FIELDS + ["chain_hash"])
            writer.writeheader()
            # Write metadata as comment
            f.write(f"# SOC-2 Export Metadata: {json.dumps(export_metadata)}\n")
            for record in records:
                writer.writerow(record)
    
    await client.close()
    
    print(f"✅ Exported {export_metadata['record_count']} records to {output_path}.{output_format}")
    print(f"🔐 Export ID: {export_metadata['export_id']}")
    if signer_key:
        print(f"✍️  Manifest signed with HMAC-SHA256")
    
    return export_metadata

if __name__ == "__main__":
    # Usage: python export_audit_soc2.py --start 2026-01-01 --end 2026-05-30 --format jsonl --output /secure/audit_2026Q2
    import argparse
    parser = argparse.ArgumentParser(description="SOC-2 Audit Log Export")
    parser.add_argument("--start", required=True, help="Start date (ISO 8601)")
    parser.add_argument("--end", required=True, help="End date (ISO 8601)")
    parser.add_argument("--format", choices=["jsonl", "csv"], default="jsonl")
    parser.add_argument("--output", required=True, help="Output path (without extension)")
    parser.add_argument("--signer-key", help="HMAC key for manifest signing (optional but recommended)")
    
    args = parser.parse_args()
    
    asyncio.run(export_audit_logs(
        start_date=args.start,
        end_date=args.end,
        output_format=args.format,
        output_path=args.output,
        signer_key=args.signer_key
    ))
```

### C. Access Review Report Generator (`scripts/generate_access_report.py`)

```python
#!/usr/bin/env python3
"""
Generate quarterly access review report for SOC-2 compliance.
Lists all users with permissions, station assignments, and last activity.
"""
import asyncio
import csv
from datetime import datetime, timedelta
from motor.motor_asyncio import AsyncIOMotorClient
from backend.core import MONGO_URL, now_iso

async def generate_access_report(output_path: str = "access_review.csv"):
    client = AsyncIOMotorClient(MONGO_URL)
    db = client.get_database()
    
    # Fetch all non-system users with permissions
    users_cursor = db.users.find(
        {"role": {"$nin": ["system", "migration"]}},
        {
            "id": 1, "email": 1, "role": 1, "tier": 1,
            "assigned_stations": 1, "permissions": 1,
            "last_login": 1, "created_at": 1
        }
    )
    
    # Fetch recent audit activity for each user (last 90 days)
    cutoff = (datetime.now() - timedelta(days=90)).isoformat()
    
    with open(output_path, "w", newline="") as f:
        writer = csv.writer(f)
        # SOC-2 required headers for access reviews
        writer.writerow([
            "User ID", "Email", "Role", "Tier", "Assigned Stations",
            "Permission Count", "Last Login", "Account Created",
            "Recent Actions (90d)", "Review Status", "Reviewer Notes"
        ])
        
        async for user in users_cursor:
            # Count recent activity from audit logs
            activity_count = await db.audit_logs_soc2.count_documents({
                "user_id": user["id"],
                "timestamp": {"$gte": cutoff}
            })
            
            # Format stations list
            stations = ", ".join(user.get("assigned_stations", [])) or "NONE"
            
            # Permission count (from cached array or compute)
            perm_count = len(user.get("permissions", []))
            
            writer.writerow([
                user["id"],
                user["email"],
                user["role"],
                user["tier"],
                stations,
                perm_count,
                user.get("last_login", "NEVER"),
                user.get("created_at", "UNKNOWN"),
                f"{activity_count} events",
                "PENDING_REVIEW",  # Default status
                ""  # Notes for reviewer
            ])
    
    await client.close()
    
    # Generate summary metadata
    summary = {
        "report_id": f"AR-{datetime.now().strftime('%YQ%m')}",
        "generated_at": now_iso(),
        "review_period": "90 days",
        "total_users": await db.users.count_documents({"role": {"$nin": ["system", "migration"]}}),
        "output_file": output_path,
        "next_review_due": (datetime.now() + timedelta(days=90)).isoformat()
    }
    
    # Write summary alongside CSV
    with open(output_path.replace(".csv", ".metadata.json"), "w") as f:
        import json
        json.dump(summary, f, indent=2)
    
    print(f"✅ Access review report generated: {output_path}")
    print(f"📋 Summary: {summary['total_users']} users, review due {summary['next_review_due']}")
    return summary

if __name__ == "__main__":
    asyncio.run(generate_access_report())
```
