# Server-side pepper for password hashing (SOC-2 CC3.1)
PASSWORD_PEPPER = os.environ.get("PASSWORD_PEPPER", "")

def _hash_pw(pw: str) -> str:
    """Hash password with bcrypt + server-side pepper"""
    if PASSWORD_PEPPER:
        pw = pw + PASSWORD_PEPPER
    return pwd_ctx.hash(pw)

def _verify_pw(pw: str, hashed: str) -> bool:
    """Verify password with pepper-aware comparison (constant-time)"""
    try:
        if PASSWORD_PEPPER:
            # Try with pepper first
            if pwd_ctx.verify(pw + PASSWORD_PEPPER, hashed):
                return True
            # Fallback for legacy hashes without pepper
            return pwd_ctx.verify(pw, hashed)
        return pwd_ctx.verify(pw, hashed)
    except Exception:
        return False

# JWT Token Revocation Support
async def is_token_revoked(jti: str) -> bool:
    """Check if token ID is in revocation list"""
    return await db.revoked_tokens.find_one({

def _make_token(user_id: str, extra_claims: Optional[dict] = None) -> str:
    """Generate JWT with SOC-2 compliant claims"""
    now = datetime.now(timezone.utc)
    exp = now + timedelta(hours=JWT_EXPIRE_HOURS)
    payload = {
        "sub": user_id,
        "iss": "fuelpro-backend",      # Token issuer
        "aud": "fuelpro-client",        # Intended audience
        "exp": exp,
        "iat": now,
        "jti": new_id(),                # Unique ID for revocation
        "scope": "user",                # Default scope
        **(extra_claims or {}),
    }
    return jwt.encode(payload, JWT_SECRET, algorithm=JWT_ALG)

# Update get_current_user to check revocation
async def get_current_user(
    creds: Optional[HTTPAuthorizationCredentials] = Depends(bearer),
) -> dict:
    if not creds:
        raise HTTPException(status_code=401, detail="Missing token")
    try:
        payload = jwt.decode(creds.credentials, JWT_SECRET, algorithms=[JWT_ALG])
        uid = payload.get("sub")
        jti = payload.get("jti")  # Token unique ID
        
        # CRITICAL: Check if token was revoked
        if jti and await is_token_revoked(jti):
            raise HTTPException(status_code=401, detail="Token revoked")
    except JWTError as e:
        raise HTTPException(status_code=401, detail=f"Invalid token: {e}")
    
    if not uid:
        raise HTTPException(status_code=401, detail="Invalid token payload")
    
    user = await db.users.find_one({"id": uid}, {"_id": 0})
    if not user:
        raise HTTPException(status_code=401, detail="User not found")
    return user

async def mpesa_stk_callback_handler(request: Request):
    """Handle MPesa STK Push callback with signature verification"""
    # Read raw body for signature verification
    body_bytes = await request.body()
    
    # Extract signature headers
    x_signature = request.headers.get("X-Signature")
    x_timestamp = request.headers.get("X-Timestamp")
    
    # Verify signature if provided (production requirement)
    if IS_PRODUCTION and x_signature and x_timestamp:
        expected_sig = hmac.new(
            MPESA_WEBHOOK_SECRET.encode(),
            body_bytes + x_timestamp.encode(),
            hashlib.sha256
        ).hexdigest()
        
        if not hmac.compare_digest(expected_sig, x_signature):
            log.warning("MPesa webhook signature mismatch from %s", request.client.host)
            # Still return 200 to prevent Safaricom retry storms
            return {"ResultCode": 0, "ResultDesc": "Callback received (unverified)"}
    
    # Parse callback
    import json
    try:
        payload = json.loads(body_bytes)
    except json.JSONDecodeError:
        log.error("Invalid JSON in MPesa callback")
        return {"ResultCode": 1, "ResultDesc": "Invalid JSON"}
    
    parsed = _parse_stk_callback(payload)
    log.info(
        "fuelpro.mpesa.callback.parsed checkout_id=%s result_code=%s status=%s receipt=%s",
        parsed.get("checkout_request_id"), parsed.get("result_code"),
        parsed.get("status_str"), parsed.get("receipt"),
    )
    
    tx = await db.payment_transactions.find_one(
        {"checkout_request_id": parsed["checkout_request_id"]}, {"_id": 0},
    )
    
    if tx and tx.get("payment_status") != "paid":
        now = now_iso()
        await _update_payment_transaction(parsed, now)

            await _activate_subscription_from_callback(tx, parsed, now)
    
    # Log to audit trail with signature verification status
    await db.audit_log.insert_one({
        "id": new_id(),
        "action": "mpesa.callback_received",
        "resource_type": "payment_webhook",
        "ip_address": request.client.host if request.client else "unknown",
        "timestamp": now,
        "meta": {
            "callback_type": "stk",
            "merchant_request_id": parsed.get("merchant_request_id"),
            "signature_verified": bool(x_signature and x_timestamp),
        },
    })

// === NEW FILE: frontend/src/lib/locationService.ts ===
export interface UserLocation {
  country: string;
  country_code: string;
  region: string;
  city: string;
  latitude: number;
  longitude: number;
  timezone: string;
  currency: string;
  language: string;
  ip: string;
}

export interface LocationConfig {
  currency: string;
  language: string;
  units: 'metric' | 'imperial';
  paymentMethods: string[];
  taxRate: number;
  complianceRules: string[];
  fuelPriceAPI: string;
  newsSources: string[];
  localMedia: LocalMediaConfig[];
  legislation: LegislationConfig[];
}

export interface LocalMediaConfig {
  type: 'tv' | 'radio' | 'magazine' | 'blogger' | 'social';
  name: string;
  url: string;
  language: string;
  category: string;
}

export interface LegislationConfig {
  title: string;
  type: 'bill' | 'law' | 'regulation' | 'ruling';
  status: 'pending' | 'active' | 'archived';
  effectiveDate: string;
  description: string;
  url: string;
}

class LocationService {
  private location: UserLocation | null = null;
  private config: LocationConfig | null = null;
  private CACHE_KEY = 'fuelpro_user_location';
  private CACHE_DURATION = 30 * 60 * 1000; // 30 minutes

  async detectLocation(): Promise<UserLocation> {
    // Check cache first
    const cached = this.getCachedLocation();
    if (cached) {
      this.location = cached;
      return cached;
    }

    try {
      // Try browser geolocation first (most accurate)
      if ('geolocation' in navigator) {
        const position = await this.getBrowserGeolocation();
        await this.reverseGeocode(position.coords.latitude, position.coords.longitude);
      }

      // Fallback to IP-based geolocation
      const response = await fetch('https://ipapi.co/json/');
      const data = await response.json();
      
      this.location = {
        country: data.country_name || 'United States',
        country_code: data.country_code || 'US',
        region: data.region || '',
        city: data.city || '',
        latitude: data.latitude || 0,
        longitude: data.longitude || 0,
        timezone: data.timezone || 'UTC',
        currency: data.currency || 'USD',
        language: this.detectLanguage(data.languages),
        ip: data.ip
      };
      
      this.cacheLocation(this.location);
      return this.location;
    } catch (error) {
      console.error('Location detection failed:', error);
      // Default fallback
      this.location = {
        country: 'United States',
        country_code: 'US',
        region: '',
        city: '',
        latitude: 0,
        longitude: 0,
        timezone: 'UTC',
        currency: 'USD',
        language: 'en',
        ip: ''
      };
      return this.location;
    }
  }

  private getBrowserGeolocation(): Promise<GeolocationPosition> {
    return new Promise((resolve, reject) => {
      navigator.geolocation.getCurrentPosition(
        resolve,
        reject,
        {
          enableHighAccuracy: true,
          timeout: 10000,
          maximumAge: 0
        }
      );
    });
  }

  private async reverseGeocode(lat: number, lon: number): Promise<void> {
    try {
      const response = await fetch(
        `https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lon}`
      );
      const data = await response.json();
      // Update location with detailed info
      if (this.location) {
        this.location.city = data.address?.city || data.address?.town || '';
        this.location.region = data.address?.state || data.address?.county || '';
        this.location.country = data.address?.country || this.location.country;
        this.location.country_code = data.address?.country_code || this.location.country_code;
      }
    } catch (error) {
      console.error('Reverse geocoding failed:', error);
    }
  }

  private detectLanguage(languages?: string): string {
    if (!languages) return 'en';
    const primaryLang = languages.split(',')[0].split('-')[0];
    return primaryLang || 'en';
  }

  async loadLocationConfig(): Promise<LocationConfig> {
    if (!this.location) {
      await this.detectLocation();
    }
    // Load country-specific configuration from backend
    const config = await this.fetchCountryConfig(this.location!.country_code);
    this.config = config;
    return config;
  }

  private async fetchCountryConfig(countryCode: string): Promise<LocationConfig> {
    // Fetch from your backend endpoint
    try {
      const response = await fetch(`/api/location/config/${countryCode}`);
      if (response.ok) {
        return await response.json();
      }
    } catch (error) {
      console.warn('Failed to fetch country config, using defaults');
    }
    
    // Default configuration
    return {
      currency: 'USD',
      language: 'en',
      units: 'metric',
      paymentMethods: ['stripe', 'paypal'],
      taxRate: 0.0,
      complianceRules: [],
      fuelPriceAPI: '',
      newsSources: [],
      localMedia: [],
      legislation: []
    };
  }

  private getCachedLocation(): UserLocation | null {
    try {
      const cached = localStorage.getItem(this.CACHE_KEY);
      if (!cached) return null;
      const parsed = JSON.parse(cached);
      const age = Date.now() - parsed.timestamp;
      if (age > this.CACHE_DURATION) {
        localStorage.removeItem(this.CACHE_KEY);
        return null;
      }
      return parsed.data;
    } catch {
      return null;
    }
  }

  private cacheLocation(location: UserLocation): void {
    try {
      localStorage.setItem(this.CACHE_KEY, JSON.stringify({
        timestamp: Date.now(),
        data: location
      }));
    } catch (error) {
      console.error('Failed to cache location:', error);
    }
  }

  // Helper getters
  getCurrency(): string {
    return this.config?.currency || this.location?.currency || 'USD';
  }
  
  getLanguage(): string {
    return this.config?.language || this.location?.language || 'en';
  }
  
  getUnits(): 'metric' | 'imperial' {
    return this.config?.units || 'metric';
  }
  
  getPaymentMethods(): string[] {
    return this.config?.paymentMethods || ['stripe'];
  }
  
  getLocation(): UserLocation | null {
    return this.location;
  }
}

export const locationService = new LocationService();

// === NEW FILE: frontend/src/lib/fileUtils.ts ===

/**
 * Universal file download trigger for mobile/desktop
 * Works on iOS Safari, Android Chrome, and all desktop browsers
 */
export function triggerDownload(blob: Blob, filename: string): void {
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.style.display = 'none';
  
  // iOS Safari workaround: must be called on user event
  document.body.appendChild(a);
  a.click();
  
  // Cleanup
  setTimeout(() => {
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  }, 100);
}

/**
 * Mobile-safe PDF viewer/download
 */
export async function handlePDFDownload(pdfUrl: string, filename: string): Promise<void> {
  const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent);
  
  try {
    const response = await fetch(pdfUrl);
    const blob = await response.blob();
    
    if (isIOS) {
      // iOS: Open in new tab (more reliable than download)
      const url = URL.createObjectURL(blob);
      window.open(url, '_blank');
      setTimeout(() => URL.revokeObjectURL(url), 1000);
    } else {
      // Android/Desktop: Trigger download
      triggerDownload(blob, filename);
    }
  } catch (error) {
    console.error('PDF download failed:', error);
    throw new Error('Failed to download PDF. Please try again.');
  }
}

@router.get("/detect", response_model=LocationResponse)
async def detect_location(request: Request):
    """Detect user location from IP or provided coordinates"""
    client_ip = request.client.host if request.client else "127.0.0.1"
    
    try:
        async with httpx.AsyncClient() as client:
            response = await client.get(f"https://ipapi.co/{client_ip}/json/")
            data = response.json()
            
            return LocationResponse(
                country=data.get("country_name", "United States"),
                country_code=data.get("country_code", "US"),
                region=data.get("region", ""),
                city=data.get("city", ""),
                latitude=data.get("latitude", 0),
                longitude=data.get("longitude", 0),
                timezone=data.get("timezone", "UTC"),
                currency=data.get("currency", "USD"),
                language=data.get("languages", "en").split(",")[0].split("-")[0]
            )
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Location detection failed: {str(e)}")
