```python
from google.oauth2 import id_token
from google.auth.transport import requests as google_requests

class GoogleAuthBody(BaseModel):
    id_token: str = Field(..., min_length=100, description="Google ID token from client")
    platform: Literal["web", "android", "ios"] = "web"

@router.post("/auth/google", response_model=TokenResponse)
async def google_auth_exchange(body: GoogleAuthBody, request: Request):
    """
    Verify Google ID token directly and create FuelPro session.
    SOC-2 compliant: token verified server-side, no proxy dependency.
    """
    # Validate platform client ID exists
    client_id = GOOGLE_CLIENT_IDS.get(body.platform)
    if not client_id or not is_google_client_id_valid(client_id):
        log.error("Google client ID misconfigured for platform: %s", body.platform)
        raise HTTPException(
            status_code=500,
            detail="Authentication service misconfigured",
            headers={"X-Error-Code": "GOOGLE_CONFIG_INVALID"}
        )
    
    try:
        # Verify token signature, expiration, audience, issuer with Google's public keys
        token_info = id_token.verify_oauth2_token(
            body.id_token,
            google_requests.Request(),
            audience=client_id,
            clock_skew_in_seconds=10  # Allow minor clock drift
        )
        
        # Validate required claims
        if not token_info.get("email") or not token_info.get("email_verified"):
            raise ValueError("Token missing verified email")
            
    except ValueError as e:
        # Log detailed error for debugging (never expose to client)
        log.warning("Google token verification failed: %s | token_prefix=%s",

        raise HTTPException(
            status_code=401,
            detail="Invalid authentication token",
            headers={"WWW-Authenticate": "Bearer error=invalid_token"}
        )
    except Exception as e:
        log.error("Google auth service error: %s", str(e))
        raise HTTPException(
            status_code=503,
            detail="Authentication service temporarily unavailable",
            headers={"Retry-After": "60"}
        )
    
    # Upsert user (same logic as before, but with enhanced audit)
    email = token_info["email"].lower().strip()
    now = datetime.now(timezone.utc)
    
    existing = await db.users.find_one({"email": email}, {"_id": 0})
    
    if existing:
        # Update existing user with Google profile data
        await db.users.update_one(
            {"email": email},
            {"$set": {
                "name": token_info.get("name", existing.get("name")),
                "google_picture": token_info.get("picture"),
                "auth_methods": list(set((existing.get("auth_methods") or []) + ["google"])),
                "last_oauth_at": now.isoformat(),
                "updated_at": now.isoformat(),
                "email_verified": True  # Google-verified

        user_doc = await db.users.find_one({"email": email}, {"_id": 0})
    else:

            "id": new_id(),
            "email": email,
            "name": token_info.get("name", email.split("@")[0]),
            "password_hash": "",  # No password for OAuth-only users
            "role": "owner",
            "tier": "free",

            "trial_started_at": now.isoformat(),

            "google_picture": token_info.get("picture"),
            "google_sub": token_info["sub"],  # Store Google ID for linking
            "auth_methods": ["google"],
            "email_verified": True,
            "created_at": now.isoformat(),
            "updated_at": now.isoformat(),
        }
        await db.users.insert_one(user_doc)
    
    # SOC-2 audit: Log authentication event with full context
    await db.audit_log.insert_one({
        "id": new_id(),
        "user_id": user_doc["id"],
        "action": "auth.google_signin",
```

```python
def _make_token(user_id: str, extra_claims: Optional[dict] = None) -> str:
    """
    Generate JWT with SOC-2 compliant claims.
    Includes: sub, iss, aud, exp, iat, jti, scope, auth_provider
    """
    now = datetime.now(timezone.utc)
    exp = now + timedelta(hours=JWT_EXPIRE_HOURS)
    
    payload = {
        "sub": user_id,
        "iss": "fuelpro-backend",  # Token issuer
        "aud": "fuelpro-client",   # Intended audience

        "iat": now,
        "jti": new_id(),  # Unique token ID for revocation
```

```typescript
// === CREATE FILE: frontend/src/lib/googleAuth.ts ===

import { Capacitor } from '@capacitor/core';
import { GoogleAuth } from '@codetrix-studio/capacitor-google-auth';

const GOOGLE_CONFIG = {
  web: { clientId: import.meta.env.VITE_GOOGLE_CLIENT_ID_WEB },
  android: { clientId: import.meta.env.VITE_GOOGLE_CLIENT_ID_ANDROID },
  ios: { clientId: import.meta.env.VITE_GOOGLE_CLIENT_ID_IOS },
};

export async function signInWithGoogle(platform: 'web' | 'android' | 'ios' = 'web') {
  const config = GOOGLE_CONFIG[platform];
  
  if (!config.clientId) {
    throw new Error(`Google client ID not configured for ${platform}`);
  }

  if (Capacitor.isNativePlatform()) {
    // Native mobile: use Capacitor plugin
    await GoogleAuth.initialize({
      clientId: config.clientId,
      scopes: ['email', 'profile'],
      grantOfflineAccess: false,
    });
    
    const result = await GoogleAuth.signIn();
    
    if (!result?.authentication?.idToken) {
      throw new Error('Google Sign-In returned no ID token');
    }
    
    return {
      idToken: result.authentication.idToken,
      email: result.email,
      name: result.name,
      photoUrl: result.imageUrl,
    };
  } else {
    // Web: use Google Identity Services
    return new Promise((resolve, reject) => {
      // @ts-ignore - google.accounts loaded via script tag in index.html
      const client = google.accounts.oauth2.initTokenClient({
        client_id: config.clientId,
        scope: 'email profile',
        prompt: 'select_account',
        callback: (response: any) => {
          if (response.error) {
            reject(new Error(`Google OAuth: ${response.error}`));
            return;
          }
          // Fetch profile + return ID token
          fetch('https://www.googleapis.com/oauth2/v3/userinfo', {
            headers: { Authorization: `Bearer ${response.access_token}` }
          })
          .then(res => res.json())
          .then(profile => resolve({
            idToken: response.id_token,
            email: profile.email,
            name: profile.name,
            photoUrl: profile.picture,
          }))
          .catch(reject);
        },
      });
      client.requestAccessToken();
    });
  }
}
```

```python
# Server-side pepper for password hashing (SOC-2 CC3.1)
# Rotate quarterly; store in AWS Secrets Manager in production
PASSWORD_PEPPER = os.environ.get("PASSWORD_PEPPER", "")

def _hash_pw(pw: str) -> str:
    """Hash password with bcrypt + server-side pepper"""
    if PASSWORD_PEPPER:
        pw = pw + PASSWORD_PEPPER
    return pwd_ctx.hash(pw)

def _verify_pw(pw: str, hashed: str) -> bool:
    """Verify password with pepper-aware comparison"""
    try:
        if PASSWORD_PEPPER:
            # Try with pepper first
            if pwd_ctx.verify(pw + PASSWORD_PEPPER, hashed):
                return True
            # Fallback: try without pepper (for legacy hashes)
            return pwd_ctx.verify(pw, hashed)
        return pwd_ctx.verify(pw, hashed)
    except Exception:
        return False
```

```python
    x_signature: Optional[str] = Header(None, alias="X-Signature"),
    x_timestamp: Optional[str] = Header(None, alias="X-Timestamp"),
):
    """
    Handle MPesa STK Push callback with signature verification.
    Prevents replay attacks and unauthorized webhook calls.
    """
    # Read raw body for signature verification
    body_bytes = await request.body()
    
    # Verify signature if provided (production requirement)
    if IS_PRODUCTION and x_signature and x_timestamp:
```

```python
from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request

class RequestIDMiddleware(BaseHTTPMiddleware):
    """Add X-Request-ID to all requests/responses for distributed tracing"""
    async def dispatch(self, request: Request, call_next):
        request_id = request.headers.get("x-request-id") or new_id()
        request.state.request_id = request_id
        
        response = await call_next(request)
        response.headers["X-Request-ID"] = request_id
        return response

app.add_middleware(RequestIDMiddleware)
```

```python
async def with_timeout(coro, timeout_seconds: int = None):
    """Execute coroutine with timeout, raising TimeoutError if exceeded"""
    import asyncio
    timeout = timeout_seconds or DEFAULT_TIMEOUT_SECONDS
    return await asyncio.wait_for(coro, timeout=timeout)
```
